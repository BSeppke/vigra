00048 template<class GRAPH, class COPYFROM, class COPYTO, class FUNCTOR>
00049 void
00050 transform_vertex_property_map(const GRAPH &graph, 
00051                               const COPYFROM &from,
00052                               COPYTO &to,
00053                               FUNCTOR functor) 
00054 {
00055     typedef typename graph_traits<GRAPH>::vertex_iterator vertex_iterator;
00056     vertex_iterator i,ie;
00057     tie(i, ie) = vertices(graph);
00058     for (; i != ie; ++i) {
00059         put(to, *i, functor(get(from, *i)));
00060     }
00061 }
00062 
00063 template<class GRAPH, class COPYFROM, class COPYTO> 
00064 void
00065 copy_vertex_property_map(const GRAPH &graph, 
00066                          const COPYFROM &from,
00067                          COPYTO &to) 
00068 {
00069     transform_vertex_property_map(graph, from, to, vigra::functor::Arg1());
00070 }

00074 template<class GRAPH, class COPYFROM, class COPYTO, class FUNCTOR>
00075 void
00076 transform_edge_property_map(const GRAPH &graph, 
00077                             const COPYFROM &from,
00078                             COPYTO &to,
00079                             FUNCTOR functor) 
00080 {
00081     typedef typename graph_traits<GRAPH>::edge_iterator edge_iterator;
00082     edge_iterator i,ie;
00083     tie(i, ie) = edges(graph);
00084     size_t count = 0;
00085     for (; i != ie; ++i, ++count) {
00086         std::cout << " iter edge " << *i << std::endl;
00087         put(to, *i, functor(get(from, *i)));
00088     }
00089     std::cout << " PROCESSED " << count << " EDGES." << std::endl;
00090 }
