00137     /* generates the stride for a gapless shape.
00138     */
00139 template <int N>
00140 inline TinyVector <MultiArrayIndex, N>
00141 defaultStride(const TinyVector <MultiArrayIndex, N> &shape)
00142 {
00143     TinyVector <MultiArrayIndex, N> ret;
00144     ret [0] = 1;
00145     for (int i = 1; i < (int)N; ++i)
00146         ret [i] = ret [i-1] * shape [i-1];
00147     return ret;
00148 }

00150     /* generates the stride for a gapless shape.
00151     */
00152 template <int N>
00153 inline TinyVector <MultiArrayIndex, N>
00154 defaultMultibandStride(const TinyVector <MultiArrayIndex, N> &shape)
00155 {
00156     TinyVector <MultiArrayIndex, N> ret;
00157     ret [N-1] = 1;
00158     for (int i = 0; i < (int)N-1; ++i)
00159     {
00160         int j = (i + int(N - 1)) % N;
00161         ret [i] = ret [j] * shape [j];
00162     }
00163     return ret;
00164 }

00443 // a border type is a compact bit-wise encoding of the fact that a 
00444 // given coordinate is at the border of the ROI. Each border corresponds
00445 // to one bit in the encoding, e.g. the left, right, top, bottom borders
00446 // of a 2D image are represented by bits 0 to 3 respectively. 
00447 // If a bit is set, the point in question is at the corresponding border.
00448 // A code of all zeros therefore means that the point is in the interior
00449 // of the ROI
00450 template <unsigned int N, unsigned int DIMENSION=N-1>
00451 struct BorderTypeImpl
00452 {
00453     typedef typename MultiArrayShape<N>::type shape_type;
00454     
00455     static unsigned int exec(shape_type const & point, shape_type const & shape)
00456     {
00457         unsigned int res = BorderTypeImpl<N, DIMENSION-1>::exec(point, shape);
00458         if(point[DIMENSION] == 0)
00459             res |= (1 << 2*DIMENSION);
00460         if(point[DIMENSION] == shape[DIMENSION]-1)
00461             res |= (2 << 2*DIMENSION);
00462         return res;
00463     }
00464 };
00465 
00466 template <unsigned int N>
00467 struct BorderTypeImpl<N, 0>
00468 {
00469     typedef typename MultiArrayShape<N>::type shape_type;
00470     static const unsigned int DIMENSION = 0;
00471     
00472     static unsigned int exec(shape_type const & point, shape_type const & shape)
00473     {
00474         unsigned int res = 0;
00475         if(point[DIMENSION] == 0)
00476             res |= (1 << 2*DIMENSION);
00477         if(point[DIMENSION] == shape[DIMENSION]-1)
00478             res |= (2 << 2*DIMENSION);
00479         return res;
00480     }

00489 // Create the offsets to all direct neighbors, starting from the given Level (=dimension)
00490 // and append them to the given array. The algorithm is designed so that the offsets are 
00491 // sorted by ascending strides. This has two important consequences:
00492 //  * The first half of the array contains the causal neighbors (negative strides),
00493 //    the second half the anti-causal ones (positive strides), where 'causal' refers
00494 //    to all scan-order predecessors of the center pixel, and 'anticausal' to its successors.
00495 //  * For any neighbor k, its opposite (=point-reflected) neighbor is located at index
00496 //    'N-1-k', where N is the total number of neighbors.
00497 // The function 'exists' returns an array of flags that contains 'true' when the corresponding 
00498 // neighbor is inside the ROI for the given borderType, 'false' otherwise. 
00499 template <unsigned int Level>
00500 struct MakeDirectArrayNeighborhood
00501 {
00502     template <class Array>
00503     static void offsets(Array & a)
00504     {
00505         typedef typename Array::value_type Shape;
00506         
00507         Shape point;
00508         point[Level] = -1;
00509         a.push_back(point);
00510         MakeDirectArrayNeighborhood<Level-1>::offsets(a);
00511         point[Level] = 1;
00512         a.push_back(point);
00513     }
00514     
00515     template <class Array>
00516     static void exists(Array & a, unsigned int borderType)
00517     {
00518         a.push_back((borderType & (1 << 2*Level)) == 0);
00519         MakeDirectArrayNeighborhood<Level-1>::exists(a, borderType);
00520         a.push_back((borderType & (2 << 2*Level)) == 0);
00521     }
00522 };
00523 
00524 template <>
00525 struct MakeDirectArrayNeighborhood<0>
00526 {
00527     template <class Array>
00528     static void offsets(Array & a)
00529     {
00530         typedef typename Array::value_type Shape;
00531         
00532         Shape point;
00533         point[0] = -1;
00534         a.push_back(point);
00535         point[0] = 1;
00536         a.push_back(point);
00537     }
00538     
00539     template <class Array>
00540     static void exists(Array & a, unsigned int borderType)
00541     {
00542         a.push_back((borderType & 1) == 0);
00543         a.push_back((borderType & 2) == 0);
00544     }
00545 };

00547 // Likewise, create the offsets to all indirect neighbors according to the same rules.
00548 template <unsigned int Level>
00549 struct MakeIndirectArrayNeighborhood
00550 {
00551     template <class Array, class Shape>
00552     static void offsets(Array & a, Shape point, bool isCenter = true)
00553     {
00554         point[Level] = -1;
00555         MakeIndirectArrayNeighborhood<Level-1>::offsets(a, point, false);
00556         point[Level] = 0;
00557         MakeIndirectArrayNeighborhood<Level-1>::offsets(a, point, isCenter);
00558         point[Level] = 1;
00559         MakeIndirectArrayNeighborhood<Level-1>::offsets(a, point, false);
00560     }
00561     
00562     template <class Array>
00563     static void exists(Array & a, unsigned int borderType, bool isCenter = true)
00564     {
00565         if((borderType & (1 << 2*Level)) == 0)
00566             MakeIndirectArrayNeighborhood<Level-1>::exists(a, borderType, false);
00567         else 
00568             MakeIndirectArrayNeighborhood<Level-1>::markOutside(a);
00569 
00570         MakeIndirectArrayNeighborhood<Level-1>::exists(a, borderType, isCenter);
00571 
00572         if((borderType & (2 << 2*Level)) == 0)
00573             MakeIndirectArrayNeighborhood<Level-1>::exists(a, borderType, false);
00574         else
00575             MakeIndirectArrayNeighborhood<Level-1>::markOutside(a);
00576     }

00580     {
00581         // Call markOutside() three times, for each possible offset at (Level-1)
00582         MakeIndirectArrayNeighborhood<Level-1>::markOutside(a);
00583         MakeIndirectArrayNeighborhood<Level-1>::markOutside(a);
00584         MakeIndirectArrayNeighborhood<Level-1>::markOutside(a);

00628 // Create the list of neighbor offsets for the given neighborhood type 
00629 // and dimension (the dimension is implicitly defined by the Shape type)
00630 // an return it in 'neighborOffsets'. Moreover, create a list of flags
00631 // for each BorderType that is 'true' when the corresponding neighbor exists
00632 // in this border situation and return the result in 'neighborExists'.
00633 template <class Shape>
00634 void
00635 makeArrayNeighborhood(ArrayVector<Shape> & neighborOffsets, 
00636                       ArrayVector<ArrayVector<bool> > & neighborExists,
00637                       NeighborhoodType neighborhoodType = DirectNeighborhood)
00638 {
00639     enum { N = Shape::static_size };
00640     
00641     neighborOffsets.clear();
00642     if(neighborhoodType == DirectNeighborhood)
00643     {
00644         MakeDirectArrayNeighborhood<N-1>::offsets(neighborOffsets);
00645     }
00646     else
00647     {
00648         Shape point; // represents the center
00649         MakeIndirectArrayNeighborhood<N-1>::offsets(neighborOffsets, point);
00650     }
00651     
00652     unsigned int borderTypeCount = 1 << 2*N;
00653     neighborExists.resize(borderTypeCount);
00654 
00655     for(unsigned int k=0; k<borderTypeCount; ++k)
00656     {
00657         neighborExists[k].clear();
00658         if(neighborhoodType == DirectNeighborhood)
00659         {
00660             MakeDirectArrayNeighborhood<N-1>::exists(neighborExists[k], k);
00661         }
00662         else
00663         {
00664             MakeIndirectArrayNeighborhood<N-1>::exists(neighborExists[k], k);
00665         }
00666     }
00667 }
