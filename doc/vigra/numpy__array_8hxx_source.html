00340             to an ascending stride order.

00343     {

00347         difference_type stride(PyArray_STRIDES(pyArray()), PyArray_STRIDES(pyArray()) + N),
00348                         permutation(N);
00349         for(MultiArrayIndex k=0; k<N; ++k)
00350             permutation[k] = k;
00351         for(MultiArrayIndex k=0; k<N-1; ++k)
00352         {
00353             MultiArrayIndex smallest = k;
00354             for(MultiArrayIndex j=k+1; j<N; ++j)
00355             {
00356                 if(stride[j] < stride[smallest])
00357                     smallest = j;
00358             }
00359             if(smallest != k)
00360             {
00361                 std::swap(stride[k], stride[smallest]);
00362                 std::swap(permutation[k], permutation[smallest]);
00363             }
00364         }
00365         difference_type ordering(N);
00366         for(MultiArrayIndex k=0; k<N; ++k)
00367             ordering[permutation[k]] = k;
00368         return ordering;
00369     }

00506 template <class TYPECODE> // pseudo-template to avoid inline expansion of the function
00507                           // will always be NPY_TYPES
00508 PyObject *
00509 constructArray(TaggedShape tagged_shape, TYPECODE typeCode, bool init, python_ptr arraytype)
00510 {
00511     ArrayVector<npy_intp> shape = finalizeTaggedShape(tagged_shape);
00512     PyAxisTags axistags(tagged_shape.axistags);
00513 
00514     int ndim = (int)shape.size();
00515     ArrayVector<npy_intp> inverse_permutation;
00516     int order = 1; // Fortran order
00517 
00518     if(axistags)
00519     {
00520         if(!arraytype)
00521             arraytype = NumpyAnyArray::getArrayTypeObject();
00522 
00523         inverse_permutation = axistags.permutationFromNormalOrder();
00524         vigra_precondition(ndim == (int)inverse_permutation.size(),
00525                      "axistags.permutationFromNormalOrder(): permutation has wrong size.");
00526     }
00527     else
00528     {
00529         arraytype = python_ptr((PyObject*)&PyArray_Type);
00530         order = 0; // C order
00531     }
00532 
00533 //    std::cerr << "constructArray: " << shape << "\n" << inverse_permutation << "\n";
00534 
00535     python_ptr array(PyArray_New((PyTypeObject *)arraytype.get(), ndim, shape.begin(),
00536                                   typeCode, 0, 0, 0, order, 0),
00537                      python_ptr::keep_count);
00538     pythonToCppException(array);
00539 
00540     if(detail::nontrivialPermutation(inverse_permutation))
00541     {
00542         PyArray_Dims permute = { inverse_permutation.begin(), ndim };
00543         array = python_ptr(PyArray_Transpose((PyArrayObject*)array.get(), &permute),
00544                            python_ptr::keep_count);
00545         pythonToCppException(array);
00546     }
00547 
00548     if(arraytype != (PyObject*)&PyArray_Type && axistags)
00549         pythonToCppException(PyObject_SetAttrString(array, "axistags", axistags.axistags) != -1);
00550 
00551     if(init)
00552         PyArray_FILLWBYTE((PyArrayObject *)array.get(), 0);
00553 
00554     return array.release();
00555 }

01118 void NumpyArray<N, T, Stride>::setupArrayView()
01119 {

01122         permutation_type permute;
01123         ArrayTraits::permutationToSetupOrder(this->pyArray_, permute);
01124 
01125         vigra_precondition(abs((int)permute.size() - actual_dimension) <= 1,
01126             "NumpyArray::setupArrayView(): got array of incompatible shape (should never happen).");
01127 
01128         applyPermutation(permute.begin(), permute.end(),
01129                          PyArray_DIMS(pyArray()), this->m_shape.begin());
01130         applyPermutation(permute.begin(), permute.end(),
01131                          PyArray_STRIDES(pyArray()), this->m_stride.begin());
01132 
01133         if((int)permute.size() == actual_dimension - 1)
01134         {
01135             this->m_shape[actual_dimension-1] = 1;
01136             this->m_stride[actual_dimension-1] = sizeof(value_type);
01137         }
01138 
01139         this->m_stride /= sizeof(value_type);
01140         this->m_ptr = reinterpret_cast<pointer>(PyArray_DATA(pyArray()));
01141         vigra_precondition(this->checkInnerStride(Stride()),
01142             "NumpyArray<..., UnstridedArrayTag>::setupArrayView(): First dimension of given array is not unstrided (should never happen).");
01143 
01144     }
