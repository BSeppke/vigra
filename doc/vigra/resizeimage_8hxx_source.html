00058 /** The Coscot interpolation function.
00059 
00060     Implements the Coscot interpolation function proposed by Maria Magnusson Seger
00061     (maria@isy.liu.se) in the context of tomographic reconstruction. It provides a fast
00062     transition between the pass- and stop-bands and minimal ripple outside the transition
00063     region. Both properties are important for this application and can be tuned by the parameters
00064     <i>m</i> and <i>h</i> (with defaults 3 and 0.5). The function is defined by
00065 
00066     \f[ f_{m,h}(x) = \left\{ \begin{array}{ll}
00067                                    \frac{1}{2m}\sin(\pi x)\cot(\pi x / (2 m))(h + (1-h)\cos(\pi x/m)) & |x| \leq m \\
00068                                   0 & \mbox{otherwise}
00069                         \end{array}\right.
00070     \f]
00071 
00072     It can be used as a functor, and as a kernel for
00073     \ref resamplingConvolveImage() to create a differentiable interpolant
00074     of an image.
00075 
00076     <b>\#include</b> <vigra/resizeimage.hxx><br>
00077     Namespace: vigra
00078 
00079     \ingroup MathFunctions
00080 */
00081 template <class T>
00082 class CoscotFunction
00083 {
00084   public:
00085 
00086         /** the kernel's value type
00087         */
00088     typedef T            value_type;
00089         /** the unary functor's argument type
00090         */
00091     typedef T            argument_type;
00092         /** the splines polynomial order
00093         */
00094     typedef T            result_type;
00095 
00096     CoscotFunction(unsigned int m = 3, double h = 0.5)
00097     : m_(m),
00098       h_(h)
00099     {}
00100 
00101         /** function (functor) call
00102         */
00103     result_type operator()(argument_type x) const
00104     {
00105         return x == 0.0 ?
00106                     1.0
00107                   : abs(x) < m_ ?
00108                         VIGRA_CSTD::sin(M_PI*x) / VIGRA_CSTD::tan(M_PI * x / 2.0 / m_) *
00109                              (h_ + (1.0 - h_) * VIGRA_CSTD::cos(M_PI * x / m_)) / 2.0 / m_
00110                       : 0.0;
00111     }
00112 
00113         /** index operator -- same as operator()
00114         */
00115     value_type operator[](value_type x) const
00116         { return operator()(x); }
00117 
00118         /** Radius of the function's support.
00119             Needed for  \ref resamplingConvolveImage(), equals m.
00120         */
00121     double radius() const
00122         { return m_; }
00123 
00124         /** Derivative order of the function: always 0.
00125         */
00126     unsigned int derivativeOrder() const
00127         { return 0; }
00128 
00129         /** Prefilter coefficients for compatibility with \ref vigra::BSpline.
00130             (array has zero length, since prefiltering is not necessary).
00131         */
00132     ArrayVector<double> const & prefilterCoefficients() const
00133     {
00134         return prefilterCoefficients_;
00135     }
00136     
00137   protected:
00138     static ArrayVector<double> prefilterCoefficients_;
00139     unsigned int m_;
00140     double h_;
00141 };

00352 template <class SrcIterator, class SrcAccessor,
00353           class DestIterator, class DestAccessor>
00354 void
00355 resizeLineLinearInterpolation(SrcIterator i1, SrcIterator iend, SrcAccessor as,
00356                            DestIterator id, DestIterator idend, DestAccessor ad)
00357 {
00358     int wold = iend - i1;
00359     int wnew = idend - id;
00360 
00361     if((wold <= 1) || (wnew <= 1)) return; // oder error ?
00362 
00363     typedef
00364         NumericTraits<typename DestAccessor::value_type> DestTraits;
00365     typedef typename DestTraits::RealPromote RealPromote;
00366 
00367     ad.set(DestTraits::fromRealPromote(as(i1)), id);
00368     ++id;
00369 
00370     --iend, --idend;
00371     ad.set(DestTraits::fromRealPromote(as(iend)), idend);
00372 
00373     double dx = (double)(wold - 1) / (wnew - 1);
00374     double x = dx;
00375 
00376     for(; id != idend; ++id, x += dx)
00377     {
00378         if(x >= 1.0)
00379         {
00380             int xx = (int)x;
00381             i1 += xx;
00382             x -= (double)xx;
00383         }
00384         double x1 = 1.0 - x;
00385 
00386         ad.set(DestTraits::fromRealPromote(RealPromote(x1 * as(i1) + x * as(i1, 1))), id);
00387     }

00599 /** \brief Resize image using B-spline interpolation.
00600 
00601     The function implements separable spline interpolation algorithm described in
00602 
00603     M. Unser, A. Aldroubi, M. Eden, <i>"B-Spline Signal Processing"</i>
00604     IEEE Transactions on Signal Processing, vol. 41, no. 2, pp. 821-833 (part I),
00605     pp. 834-848 (part II), 1993.
00606 
00607     to obtain optimal interpolation quality and speed. You may pass the function
00608     a spline of arbitrary order (e.g. <TT>BSpline<ORDER, double></tt> or
00609     <TT>CatmullRomSpline<double></tt>). The default is a third order spline
00610     which gives a twice continuously differentiable interpolant.
00611     The implementation ensures that image values are interpolated rather
00612     than smoothed by first calling a recursive (sharpening) prefilter as
00613     described in the above paper. Then the actual interpolation is done
00614     using \ref resamplingConvolveLine().
00615 
00616     The range of both the input and output images (resp. regions)
00617     must be given. The input image must have a size of at
00618     least 4x4, the destination of at least 2x2. The scaling factors are then calculated
00619     accordingly. If the source image is larger than the destination, it
00620     is smoothed (band limited) using a recursive
00621     exponential filter. The source value_type (SrcAccessor::value_type) must
00622     be a linear algebra, i.e. it must support addition, subtraction,
00623     and multiplication (+, -, *), multiplication with a scalar
00624     real number and \ref NumericTraits "NumericTraits".
00625     The function uses accessors.
00626 
00627     <b> Declarations:</b>
00628 
00629     pass 2D array views:
00630     \code
00631     namespace vigra {
00632         template <class T1, class S1,
00633                   class T2, class S2,
00634                   class SPLINE>
00635         void
00636         resizeImageSplineInterpolation(MultiArrayView<2, T1, S1> const & src,
00637                                        MultiArrayView<2, T2, S2> dest,
00638                                        SPLINE const & spline = BSpline<3, double>());
00639     }
00640     \endcode
