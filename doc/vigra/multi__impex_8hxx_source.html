00111                 value_type of the target MultiArray must match the datatype stored in the file.

00830 void 
00831 exportVolume (MultiArrayView <3, T, Tag> const & volume,
00832               const VolumeExportInfo & volinfo)
00833 {
00834     if(volinfo.getFileType() == std::string("MULTIPAGE"))
00835     {
00836         char const * mode = "w";
00837         std::string compression = "LZW";
00838         if(volinfo.getCompression() != std::string())
00839             compression = volinfo.getCompression();
00840             
00841         for(MultiArrayIndex k=0; k<volume.shape(2); ++k)
00842         {
00843             ImageExportInfo info(volinfo.getFileNameBase(), mode);
00844             info.setFileType("TIFF");
00845             info.setCompression(compression.c_str());
00846             info.setPixelType(volinfo.getPixelType());
00847             detail::setRangeMapping(volume, info, typename NumericTraits<T>::isScalar());
00848             exportImage(volume.bindOuter(k), info);
00849             mode = "a";
00850         }
00851     }
00852     else
00853     {
00854         std::string name = std::string(volinfo.getFileNameBase()) + std::string(volinfo.getFileNameExt());
00855         ImageExportInfo info(name.c_str());
00856         info.setCompression(volinfo.getCompression());
00857         info.setPixelType(volinfo.getPixelType());
00858         detail::setRangeMapping(volume, info, typename NumericTraits<T>::isScalar());
00859 
00860         const unsigned int depth = volume.shape (2);
00861         int numlen = static_cast <int> (std::ceil (std::log10 ((double)depth)));
00862         for (unsigned int i = 0; i < depth; ++i)
00863         {
00864             // build the filename
00865             std::stringstream stream;
00866             stream << std::setfill ('0') << std::setw (numlen) << i;
00867             std::string name_num;
00868             stream >> name_num;
00869             std::string name = std::string(volinfo.getFileNameBase()) + name_num + std::string(volinfo.getFileNameExt());
00870 
00871             MultiArrayView <2, T, Tag> view (volume.bindOuter (i));
00872 
00873             // export the image
00874             info.setFileName(name.c_str ());
00875             exportImage(srcImageRange(view), info); 
00876         }
