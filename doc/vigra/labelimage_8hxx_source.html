00062 /** \brief Find the connected components of a segmented image.
00063 
00064     Connected components are defined as regions with uniform pixel
00065     values. Thus, <TT>T1</TT> either must be
00066     equality comparable, or a suitable EqualityFunctor must be
00067     provided that realizes the desired predicate. The
00068     destination's value type <tt>T2</tt> should be large enough to hold the labels
00069     without overflow. Region numbers will be a consecutive sequence
00070     starting with one and ending with the region number returned by
00071     the function (inclusive). The parameter '<TT>eight_neighbors</TT>'
00072     determines whether the regions should be 4-connected (false) or
00073     8-connected (true). 
00074 
00075     Return:  the number of regions found (= largest region label)
00076     
00077     See \ref labelMultiArray() for a dimension-independent implementation of 
00078     connected components labelling.

00185 template <class SrcIterator, class SrcAccessor,
00186           class DestIterator, class DestAccessor,
00187           class EqualityFunctor>
00188 unsigned int labelImage(SrcIterator upperlefts,
00189                         SrcIterator lowerrights, SrcAccessor sa,
00190                         DestIterator upperleftd, DestAccessor da,
00191                         bool eight_neighbors, EqualityFunctor equal)
00192 {
00193     typedef typename DestAccessor::value_type LabelType;
00194     
00195     int w = lowerrights.x - upperlefts.x;
00196     int h = lowerrights.y - upperlefts.y;
00197     int x,y,i;
00198 
00199     const Diff2D neighbor[] = {
00200         Diff2D(-1,0),  // left
00201         Diff2D(-1,-1), // topleft
00202         Diff2D(0,-1),  // top
00203         Diff2D(1,-1)   // topright
00204     };
00205 
00206     const int left = 0, /* unused:  topleft = 1, */ top = 2, topright = 3;
00207     int step = eight_neighbors ? 1 : 2;
00208 
00209     SrcIterator ys = upperlefts;
00210     DestIterator yd = upperleftd;
00211     
00212     detail::UnionFindArray<LabelType>  label;    
00213 
00214     // pass 1: scan image from upper left to lower right
00215     // to find connected components
00216 
00217     // Each component will be represented by a tree of pixels. Each
00218     // pixel contains the scan order address of its parent in the
00219     // tree.  In order for pass 2 to work correctly, the parent must
00220     // always have a smaller scan order address than the child.
00221     // Therefore, we can merge trees only at their roots, because the
00222     // root of the combined tree must have the smallest scan order
00223     // address among all the tree's pixels/ nodes.  The root of each
00224     // tree is distinguished by pointing to itself (it contains its
00225     // own scan order address). This condition is enforced whenever a
00226     // new region is found or two regions are merged
00227 
00228 
00229     for(y = 0; y != h; ++y, ++ys.y, ++yd.y)
00230     {
00231         SrcIterator xs = ys;
00232         DestIterator xd = yd;
00233 
00234         int endNeighbor = (y == 0) ? left : (eight_neighbors ? topright : top);
00235 
00236         for(x = 0; x != w; ++x, ++xs.x, ++xd.x)
00237         {
00238             int beginNeighbor = (x == 0) ? top : left;
00239             if(x == w-1 && endNeighbor == topright) endNeighbor = top;
00240 
00241             for(i=beginNeighbor; i<=endNeighbor; i+=step)
00242             {
00243                 if(equal(sa(xs), sa(xs, neighbor[i])))
00244                 {
00245                     LabelType neighborLabel = label.find(da(xd,neighbor[i]));
00246 
00247                     for(int j=i+2; j<=endNeighbor; j+=step)
00248                     {
00249                         if(equal(sa(xs), sa(xs, neighbor[j])))
00250                         {
00251                             neighborLabel = label.makeUnion(da(xd, neighbor[j]), neighborLabel);
00252                             break;
00253                         }
00254                     }
00255                     da.set(neighborLabel, xd);
00256                     break;
00257                 }
00258 
00259             }
00260             if(i > endNeighbor)
00261             {
00262                 da.set(label.makeNewLabel(), xd);
00263             }
00264         }
00265     }
00266 
00267     // pass 2: assign one label to each region (tree)
00268     // so that labels form a consecutive sequence 1, 2, ...
00269     unsigned int count = label.makeContiguous();    
00270     
00271     yd = upperleftd;
00272     for(y=0; y != h; ++y, ++yd.y)
00273     {
00274         typename DestIterator::row_iterator xd = yd.rowIterator();
00275         for(x = 0; x != w; ++x, ++xd)
00276         {
00277             da.set(label[da(xd)], xd);
00278         }
00279     }
00280     return count;
00281 }

00485           class DestIterator, class DestAccessor,
00486           class ValueType, class EqualityFunctor>
00487 unsigned int labelImageWithBackground(
00488     SrcIterator upperlefts,
00489     SrcIterator lowerrights, SrcAccessor sa,
00490     DestIterator upperleftd, DestAccessor da,
00491     bool eight_neighbors,
00492     ValueType background_value, EqualityFunctor equal)
00493 {
00494     int w = lowerrights.x - upperlefts.x;
00495     int h = lowerrights.y - upperlefts.y;
00496     int x,y,i;
00497 
00498     const Diff2D neighbor[] = {
00499         Diff2D(-1,0),  // left
00500         Diff2D(-1,-1), // topleft
00501         Diff2D(0,-1),  // top
00502         Diff2D(1,-1)   // topright
00503     };
00504 
00505     const int left = 0, /* unused:  topleft = 1,*/ top = 2, topright = 3;
00506     int step = eight_neighbors ? 1 : 2;
00507 
00508     SrcIterator ys(upperlefts);
00509     SrcIterator xs(ys);
00510     
00511     // temporary image to store region labels
00512     typedef BasicImage<IntBiggest> TmpImage;
00513     TmpImage labelimage(w, h);
00514     TmpImage::ScanOrderIterator label = labelimage.begin();
00515     TmpImage::Iterator yt = labelimage.upperLeft();
00516     TmpImage::Iterator  xt(yt);
00517 
00518     // pass 1: scan image from upper left to lower right
00519     // find connected components
00520 
00521     for(y = 0; y != h; ++y, ++ys.y, ++yt.y)
00522     {
00523         xs = ys;
00524         xt = yt;
00525 
00526         int endNeighbor = (y == 0) ? left : (eight_neighbors ? topright : top);
00527 
00528         for(x = 0; x != w; ++x, ++xs.x, ++xt.x)
00529         {
00530             if(equal(sa(xs), background_value))
00531             {
00532                 *xt = -1;
00533             }
00534             else
00535             {
00536                 int beginNeighbor = (x == 0) ? top : left;
00537                 if(x == w-1 && endNeighbor == topright) endNeighbor = top;
00538 
00539                 for(i=beginNeighbor; i<=endNeighbor; i+=step)
00540                 {
00541                     if(equal(sa(xs), sa(xs, neighbor[i])))
00542                     {
00543                         IntBiggest neighborLabel = xt[neighbor[i]];
00544 
00545                         for(int j=i+2; j<=endNeighbor; j+=step)
00546                         {
00547                             if(equal(sa(xs), sa(xs, neighbor[j])))
00548                             {
00549                                 IntBiggest neighborLabel1 = xt[neighbor[j]];
00550 
00551                                 if(neighborLabel != neighborLabel1)
00552                                 {
00553                                     // find roots of the region trees
00554                                     while(neighborLabel != label[neighborLabel])
00555                                     {
00556                                         neighborLabel = label[neighborLabel];
00557                                     }
00558                                     while(neighborLabel1 != label[neighborLabel1])
00559                                     {
00560                                         neighborLabel1 = label[neighborLabel1];
00561                                     }
00562 
00563                                     // merge the trees
00564                                     if(neighborLabel1 < neighborLabel)
00565                                     {
00566                                         label[neighborLabel] = neighborLabel1;
00567                                         neighborLabel = neighborLabel1;
00568                                     }
00569                                     else if(neighborLabel < neighborLabel1)
00570                                     {
00571                                         label[neighborLabel1] = neighborLabel;
00572                                     }
00573                                 }
00574                                 break;
00575                             }
00576                         }
00577                         *xt = neighborLabel;
00578                         break;
00579                     }
00580 
00581                 }
00582                 if(i > endNeighbor)
00583                 {
00584                     // new region
00585                     // The initial label of a new region equals the
00586                     // scan order address of it's first pixel.
00587                     // This is essential for correct operation of the algorithm.
00588                     *xt = x + y*w;
00589                 }
00590             }
00591         }
00592     }
00593 
00594     // pass 2: assign contiguous labels to the regions
00595     DestIterator yd(upperleftd);
00596 
00597     int count = 0;
00598     i = 0;
00599     for(y=0; y != h; ++y, ++yd.y)
00600     {
00601         DestIterator xd(yd);
00602         for(x = 0; x != w; ++x, ++xd.x, ++i)
00603         {
00604             if(label[i] == -1) continue;
00605 
00606             if(label[i] == i)
00607             {
00608                 label[i] = count++;
00609             }
00610             else
00611             {
00612                 label[i] = label[label[i]];
00613             }
00614             da.set(label[i]+1, xd);
00615         }
00616     }
00617 
00618     return count;
00619 }
