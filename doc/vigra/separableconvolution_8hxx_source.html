00060 // This function assumes that the input array is actually larger than
00061 // the range [is, iend), so that it can safely access values outside
00062 // this range. This is useful if (1) we work on a small ROI, or 
00063 // (2) we enlarge the input by copying with border treatment.
00064 template <class SrcIterator, class SrcAccessor,
00065           class DestIterator, class DestAccessor,
00066           class KernelIterator, class KernelAccessor>
00067 void internalConvolveLineOptimistic(SrcIterator is, SrcIterator iend, SrcAccessor sa,
00068                                     DestIterator id, DestAccessor da,
00069                                     KernelIterator kernel, KernelAccessor ka,
00070                                     int kleft, int kright)
00071 {
00072     typedef typename PromoteTraits<
00073             typename SrcAccessor::value_type,
00074             typename KernelAccessor::value_type>::Promote SumType;
00075 
00076     int w = std::distance( is, iend );
00077     int kw = kright - kleft + 1;
00078     for(int x=0; x<w; ++x, ++is, ++id)
00079     {
00080         SrcIterator iss = is + (-kright);
00081         KernelIterator ik = kernel + kright;
00082         SumType sum = NumericTraits<SumType>::zero();
00083 
00084         for(int k = 0; k < kw; ++k, --ik, ++iss)
00085         {
00086             sum += ka(ik) * sa(iss);
00087         }
00088 
00089         da.set(detail::RequiresExplicitCast<typename
00090                       DestAccessor::value_type>::cast(sum), id);
00091     }
00092 }

00712 {

00715     {
00716         if(w + kleft < stop)
00717             stop = w + kleft;
00718         if(start < kright)
00719         {
00720             id += kright - start;
00721             start = kright;
00722         }
00723     }
00724     else
00725     {
00726         id += kright;
00727         start = kright;
00728         stop = w + kleft;
00729     }
00730 
00731     typedef typename PromoteTraits<
00732             typename SrcAccessor::value_type,
00733             typename KernelAccessor::value_type>::Promote SumType;
00734 
00735     is += start;
00736 
00737     for(int x=start; x<stop; ++x, ++is, ++id)
00738     {
00739         KernelIterator ik = kernel + kright;
00740         SumType sum = NumericTraits<SumType>::zero();
00741 
00742         SrcIterator iss = is + (-kright);
00743         SrcIterator isend = is + (1 - kleft);
00744         for(; iss != isend ; --ik, ++iss)
00745         {
00746             sum += ka(ik) * sa(iss);
00747         }
00748 
00749         da.set(detail::RequiresExplicitCast<typename
00750                       DestAccessor::value_type>::cast(sum), id);
00751     }

02212                           unsigned int derivativeOrder,
02213                           double offset)

02216 
02217     // find kernel sum
02218     Iterator k = kernel_.begin();
02219     TmpType sum = NumericTraits<TmpType>::zero();
02220 
02221     if(derivativeOrder == 0)
02222     {
02223         for(; k < kernel_.end(); ++k)
02224         {

02245     for(; k != kernel_.end(); ++k)

02248     }

02309 template <class ARITHTYPE>
02310 void
02311 Kernel1D<ARITHTYPE>::initDiscreteGaussian(double std_dev,
02312                                           value_type norm)
02313 {
02314     vigra_precondition(std_dev >= 0.0,
02315               "Kernel1D::initDiscreteGaussian(): Standard deviation must be >= 0.");
02316 
02317     if(std_dev > 0.0)
02318     {
02319         // first calculate required kernel sizes
02320         int radius = (int)(3.0*std_dev + 0.5);
02321         if(radius == 0)
02322             radius = 1;
02323 
02324         double f = 2.0 / std_dev / std_dev;
02325 
02326         // allocate the working array
02327         int maxIndex = (int)(2.0 * (radius + 5.0 * VIGRA_CSTD::sqrt((double)radius)) + 0.5);
02328         InternalVector warray(maxIndex+1);
02329         warray[maxIndex] = 0.0;
02330         warray[maxIndex-1] = 1.0;
02331 
02332         for(int i = maxIndex-2; i >= radius; --i)
02333         {
02334             warray[i] = warray[i+2] + f * (i+1) * warray[i+1];
02335             if(warray[i] > 1.0e40)
02336             {
02337                 warray[i+1] /= warray[i];
02338                 warray[i] = 1.0;
02339             }
02340         }
02341 
02342         // the following rescaling ensures that the numbers stay in a sensible range
02343         // during the rest of the iteration, so no other rescaling is needed
02344         double er = VIGRA_CSTD::exp(-radius*radius / (2.0*std_dev*std_dev));
02345         warray[radius+1] = er * warray[radius+1] / warray[radius];
02346         warray[radius] = er;
02347 
02348         for(int i = radius-1; i >= 0; --i)
02349         {
02350             warray[i] = warray[i+2] + f * (i+1) * warray[i+1];
02351             er += warray[i];
02352         }
02353 
02354         double scale = norm / (2*er - warray[0]);
02355 
02356         initExplicitly(-radius, radius);
02357         iterator c = center();
02358 
02359         for(int i=0; i<=radius; ++i)
02360         {
02361             c[i] = c[-i] = warray[i] * scale;
02362         }
02363     }
02364     else
02365     {
02366         kernel_.erase(kernel_.begin(), kernel_.end());
02367         kernel_.push_back(norm);
02368         left_ = 0;
02369         right_ = 0;
02370     }
02371 
02372     norm_ = norm;
02373
