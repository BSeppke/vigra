00067                 typename SrcAccessor::value_type threshold,
00068                 Compare compare, AtImageBorder atBorder)
00069 {
00070     typename SrcAccessor::value_type v = sa(is);
00071     
00072     if(!compare(v, threshold))
00073         return false;
00074 
00075     int directionCount = Neighborhood::nearBorderDirectionCount(atBorder);
00076     RestrictedNeighborhoodCirculator<SrcIterator, Neighborhood> sc(is, atBorder);
00077     for(int i = 0; i < directionCount; ++i, ++sc)
00078     {
00079         if(!compare(v, sa(sc)))
00080             return false;
00081     }
00082     return true;

00597     26-neighbors in 3D, 3<sup>N</sup>-1 neighbors in N-D). 
00598     The detected points will be marked 
00599     with the default value 1 in the destination array.
00600

00602     \ref LocalMinmaxOptions : you can switch to the direct neighborhood

00605     value is not below a given threshold, allow extended minima
00606     (i.e. minima that form minimal plateaus rather than isolated pixels --
00607     note that this option is only supported for 2D images), 
00608     and change the marker in the destination image. See usage examples below 
00609     for details.
