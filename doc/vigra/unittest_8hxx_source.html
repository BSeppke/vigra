00561 template<class FPT>
00562 class close_at_tolerance {
00563 public:
00564     explicit    close_at_tolerance( FPT tolerance, bool strong_test = true )
00565         : m_strong_test( strong_test ),
00566           m_tolerance( tolerance ) {}
00567 
00568     explicit    close_at_tolerance( int number_of_rounding_errors, bool strong_test = true )
00569         : m_strong_test( strong_test ),
00570           m_tolerance( FloatTraits<FPT>::epsilon() * number_of_rounding_errors / 2.0 ) {}
00571 
00572     bool        operator()( FPT left, FPT right ) const
00573     {
00574         if (left == 0 && right != 0)
00575         {
00576                 return (fpt_abs(right) <= m_tolerance);
00577         }
00578         if (right == 0 && left != 0)
00579         {
00580                 return (fpt_abs(left) <= m_tolerance);
00581         }
00582         FPT diff = fpt_abs( left - right );
00583         FPT d1   = safe_fpt_division( diff, fpt_abs( right ) );
00584         FPT d2   = safe_fpt_division( diff, fpt_abs( left ) );
00585 
00586         return m_strong_test ? (d1 <= m_tolerance && d2 <= m_tolerance)
00587                                 : (d1 <= m_tolerance || d2 <= m_tolerance);
00588     }
00589 
00590 private:
00591     bool        m_strong_test;
00592     FPT         m_tolerance;
00593 };
