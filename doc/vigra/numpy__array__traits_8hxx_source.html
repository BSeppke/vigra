00298     {

00300         int ndim = PyArray_NDIM(array);
00301         long channelIndex = pythonGetAttr(obj, "channelIndex", ndim);

00530     template <class ARRAY>

00532     {

00534         
00535         if((int)data.size() == N)
00536         {
00537             vigra_precondition(PyArray_NDIM((PyArrayObject*)array.get()) == N,
00538                 "NumpyArray::permuteLikewise(): input array has no channel axis.");
00539 
00540             detail::getAxisPermutationImpl(permute, array, "permutationToNormalOrder", 
00541                                            AxisInfo::AllAxes, true);

00544             {
00545                 permute.resize(N);
00546                 linearSequence(permute.begin(), permute.end());
00547             }

00550                 // rotate channel axis to last position
00551                 int channelIndex = permute[0];

00553                     permute[k-1] = permute[k];
