00055 template <class Src, class Dest>
00056 Dest uninitializedCopy(Src s, Src end, Dest d)
00057 {
00058     typedef typename std::iterator_traits<Dest>::value_type T;
00059     for(; s != end; ++s, ++d)
00060         new(d) T(static_cast<T const &>(*s));
00061     return d;
00062 }
