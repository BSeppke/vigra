00048 template <class SrcIterator, class SrcAccessor,
00049           class CoeffIterator, class DestIterator>
00050 void internalNonlinearDiffusionDiagonalSolver(
00051     SrcIterator sbegin, SrcIterator send, SrcAccessor sa,
00052     CoeffIterator diag, CoeffIterator upper, CoeffIterator lower,
00053     DestIterator dbegin)
00054 {
00055     int w = send - sbegin - 1;
00056     
00057     int i;
00058     
00059     for(i=0; i<w; ++i)
00060     {
00061         lower[i] = lower[i] / diag[i];
00062         
00063         diag[i+1] = diag[i+1] - lower[i] * upper[i];
00064     }
00065     
00066     dbegin[0] = sa(sbegin);
00067     
00068     for(i=1; i<=w; ++i)
00069     {
00070         dbegin[i] = sa(sbegin, i) - lower[i-1] * dbegin[i-1];
00071     }
00072     
00073     dbegin[w] = dbegin[w] / diag[w];
00074     
00075     for(i=w-1; i>=0; --i)
00076     {
00077         dbegin[i] = (dbegin[i] - upper[i] * dbegin[i+1]) / diag[i];
00078     }
00079 }

00082 template <class SrcIterator, class SrcAccessor,
00083           class WeightIterator, class WeightAccessor,
00084           class DestIterator, class DestAccessor>
00085 void internalNonlinearDiffusionAOSStep(
00086                    SrcIterator sul, SrcIterator slr, SrcAccessor as,
00087                    WeightIterator wul, WeightAccessor aw,
00088                    DestIterator dul, DestAccessor ad, double timestep)
00089 {
00090     // use traits to determine SumType as to prevent possible overflow
00091     typedef typename
00092         NumericTraits<typename DestAccessor::value_type>::RealPromote
00093         DestType;
00094     
00095     typedef typename
00096         NumericTraits<typename WeightAccessor::value_type>::RealPromote
00097         WeightType;
00098         
00099     // calculate width and height of the image
00100     int w = slr.x - sul.x;
00101     int h = slr.y - sul.y;
00102     int d = (w < h) ? h : w;
00103 
00104     std::vector<WeightType> lower(d),
00105                             diag(d),
00106                             upper(d),
00107                             res(d);
00108 
00109     int x,y;
00110     
00111     WeightType one = NumericTraits<WeightType>::one();
00112     
00113      // create y iterators
00114     SrcIterator ys = sul;
00115     WeightIterator yw = wul;
00116     DestIterator yd = dul;
00117     
00118     // x-direction
00119     for(y=0; y<h; ++y, ++ys.y, ++yd.y, ++yw.y)
00120     {
00121         typename SrcIterator::row_iterator xs = ys.rowIterator();
00122         typename WeightIterator::row_iterator xw = yw.rowIterator();
00123         typename DestIterator::row_iterator xd = yd.rowIterator();
00124 
00125         // fill 3-diag matrix
00126         
00127         diag[0] = one + timestep * (aw(xw) + aw(xw, 1));
00128         for(x=1; x<w-1; ++x)
00129         {
00130             diag[x] = one + timestep * (2.0 * aw(xw, x) + aw(xw, x+1) + aw(xw, x-1));
00131         }
00132         diag[w-1] = one + timestep * (aw(xw, w-1) + aw(xw, w-2));
00133 
00134         for(x=0; x<w-1; ++x)
00135         {
00136             lower[x] = -timestep * (aw(xw, x) + aw(xw, x+1));
00137             upper[x] = lower[x];
00138         }
00139         
00140         internalNonlinearDiffusionDiagonalSolver(xs, xs+w, as,
00141                             diag.begin(), upper.begin(), lower.begin(), res.begin());
00142                             
00143         for(x=0; x<w; ++x, ++xd)
00144         {
00145             ad.set(res[x], xd);
00146         }
00147     }
00148         
00149     // y-direction
00150     ys = sul;
00151     yw = wul;
00152     yd = dul;
00153     
00154     for(x=0; x<w; ++x, ++ys.x, ++yd.x, ++yw.x)
00155     {
00156         typename SrcIterator::column_iterator xs = ys.columnIterator();
00157         typename WeightIterator::column_iterator xw = yw.columnIterator();
00158         typename DestIterator::column_iterator xd = yd.columnIterator();
00159 
00160         // fill 3-diag matrix
00161         
00162         diag[0] = one + timestep * (aw(xw) + aw(xw, 1));
00163         for(y=1; y<h-1; ++y)
00164         {
00165             diag[y] = one + timestep * (2.0 * aw(xw, y) + aw(xw, y+1) + aw(xw, y-1));
00166         }
00167         diag[h-1] = one + timestep * (aw(xw, h-1) + aw(xw, h-2));
00168 
00169         for(y=0; y<h-1; ++y)
00170         {
00171             lower[y] = -timestep * (aw(xw, y) + aw(xw, y+1));
00172             upper[y] = lower[y];
00173         }
00174         
00175         internalNonlinearDiffusionDiagonalSolver(xs, xs+h, as,
00176                             diag.begin(), upper.begin(), lower.begin(), res.begin());
00177                             
00178         for(y=0; y<h; ++y, ++xd)
00179         {
00180             ad.set(0.5 * (ad(xd) + res[y]), xd);
00181         }
00182     }
00183 }

00197 /** \brief Perform edge-preserving smoothing at the given scale.
00198 
00199     The algorithm solves the non-linear diffusion equation
00200     
00201     \f[
00202         \frac{\partial}{\partial t} u =
00203         \frac{\partial}{\partial x}
00204           \left( g(|\nabla u|)
00205                  \frac{\partial}{\partial x} u
00206           \right)
00207     \f]
00208     
00209     where <em> t</em> is the time, <b> x</b> is the location vector,
00210     <em> u(</em><b> x</b><em> , t)</em> is the smoothed image at time <em> t</em>, and
00211     <em> g(.)</em> is the location dependent diffusivity. At time zero, the image
00212     <em> u(</em><b> x</b><em> , 0)</em> is simply the original image. The time is
00213     proportional to the square of the scale parameter: \f$t = s^2\f$.
00214     The diffusion equation is solved iteratively according
00215     to the Additive Operator Splitting Scheme (AOS) from
00216     
00217     J. Weickert: <em>"Recursive Separable Schemes for Nonlinear Diffusion
00218     Filters"</em>,
00219     in: B. ter Haar Romeny, L. Florack, J. Koenderingk, M. Viergever (eds.):
00220         1st Intl. Conf. on Scale-Space Theory in Computer Vision 1997,
00221         Springer LNCS 1252
00222 
00223     <TT>DiffusivityFunctor</TT> implements the gradient-dependent local diffusivity.
00224     It is passed
00225     as an argument to \ref gradientBasedTransform(). The return value must be
00226     between 0 and 1 and determines the weight a pixel gets when
00227     its neighbors are smoothed. Weickert recommends the use of the diffusivity
00228     implemented by class \ref DiffusivityFunctor. It's also possible to use
00229     other functors, for example one that always returns 1, in which case
00230     we obtain the solution to the linear diffusion equation, i.e.
00231     Gaussian convolution.
00232     
00233     The source value type must be a
00234     linear space with internal addition, scalar multiplication, and
00235     NumericTraits defined. The value_type of the DiffusivityFunctor must be the
00236     scalar field over wich the source value type's linear space is defined.
00237     
00238     In addition to <TT>nonlinearDiffusion()</TT>, there is an algorithm
00239     <TT>nonlinearDiffusionExplicit()</TT> which implements the Explicit Scheme
00240     described in the above article. Both algorithms have the same interface,
00241     but the explicit scheme gives slightly more accurate approximations of
00242     the diffusion process at the cost of much slower processing.
00243     
00244     <b> Declarations:</b>

00335 template <class SrcIterator, class SrcAccessor,
00336           class DestIterator, class DestAccessor,
00337           class DiffusivityFunc>
00338 void nonlinearDiffusion(SrcIterator sul, SrcIterator slr, SrcAccessor as,
00339                    DestIterator dul, DestAccessor ad,
00340                    DiffusivityFunc const & weight, double scale)
00341 {
00342     vigra_precondition(scale > 0.0, "nonlinearDiffusion(): scale must be > 0");
00343     
00344     double total_time = scale*scale/2.0;
00345     const double time_step = 5.0;
00346     int number_of_steps = (int)(total_time / time_step);
00347     double rest_time = total_time - time_step * number_of_steps;
00348     
00349     Size2D size(slr.x - sul.x, slr.y - sul.y);
00350 
00351     typedef typename
00352         NumericTraits<typename SrcAccessor::value_type>::RealPromote
00353         TmpType;
00354     typedef typename DiffusivityFunc::value_type WeightType;
00355     
00356     BasicImage<TmpType> smooth1(size);
00357     BasicImage<TmpType> smooth2(size);
00358     
00359     BasicImage<WeightType> weights(size);
00360     
00361     typename BasicImage<TmpType>::Iterator s1 = smooth1.upperLeft(),
00362                                   s2 = smooth2.upperLeft();
00363     typename BasicImage<TmpType>::Accessor a = smooth1.accessor();
00364     
00365     typename BasicImage<WeightType>::Iterator wi = weights.upperLeft();
00366     typename BasicImage<WeightType>::Accessor wa = weights.accessor();
00367 
00368     gradientBasedTransform(sul, slr, as, wi, wa, weight);
00369 
00370     internalNonlinearDiffusionAOSStep(sul, slr, as, wi, wa, s1, a, rest_time);
00371 
00372     for(int i = 0; i < number_of_steps; ++i)
00373     {
00374         gradientBasedTransform(s1, s1+size, a, wi, wa, weight);
00375                       
00376         internalNonlinearDiffusionAOSStep(s1, s1+size, a, wi, wa, s2, a, time_step);
00377     
00378         std::swap(s1, s2);
00379     }
00380     
00381     copyImage(s1, s1+size, a, dul, ad);
00382 }

00577         gl = (aw(xw) + aw(xw, left)) * time_step;

00581         sum = g0 * as(xs);

00583         sum += gb * as(xs, top);
00584         sum += gl * as(xs, left);
00585         sum += gr * as(xs, right);

00587         ad.set(sum, xd);
00588     }

00694 /** \brief Diffusivity functor for non-linear diffusion.
00695 
00696     This functor implements the diffusivity recommended by Weickert:
00697     
00698     \f[
00699         g(|\nabla u|) = 1 -
00700            \exp{\left(\frac{-3.315}{(|\nabla u| / thresh)^4}\right)}
00701     \f]
00702     
00703     
00704     where <TT>thresh</TT> is a threshold that determines whether a specific gradient
00705     magnitude is interpreted as a significant edge (above the threshold)
00706     or as noise. It is meant to be used with \ref nonlinearDiffusion().
00707 */
00708 template <class Value>
00709 class DiffusivityFunctor
00710 {
00711   public:
00712          /** the functors first argument type (must be an algebraic field with <TT>exp()</TT> defined).
00713              However, the functor also works with RGBValue<first_argument_type> (this hack is
00714              necessary since Microsoft C++ doesn't support partial specialization).
00715          */
00716     typedef Value first_argument_type;
00717     
00718          /** the functors second argument type (same as first).
00719              However, the functor also works with RGBValue<second_argument_type> (this hack is
00720              necessary since Microsoft C++ doesn't support partial specialization).
00721          */
00722     typedef Value second_argument_type;
00723     
00724          /** the functors result type
00725          */
00726     typedef typename NumericTraits<Value>::RealPromote result_type;
00727     
00728          /** \deprecated use first_argument_type, second_argument_type, result_type
00729          */
00730     typedef Value value_type;
00731     
00732          /** init functor with given edge threshold
00733          */
00734     DiffusivityFunctor(Value const & thresh)
00735     : weight_(thresh*thresh),
00736       one_(NumericTraits<result_type>::one()),
00737       zero_(NumericTraits<result_type>::zero())
00738     {}
00739     
00740          /** calculate diffusivity from scalar arguments
00741          */
00742     result_type operator()(first_argument_type const & gx, second_argument_type const & gy) const
00743     {
00744         Value mag = (gx*gx + gy*gy) / weight_;
00745                      
00746         return (mag == zero_) ? one_ : one_ - VIGRA_CSTD::exp(-3.315 / mag / mag);
00747     }
00748     
00749          /** calculate diffusivity from RGB arguments
00750          */
00751     result_type operator()(RGBValue<Value> const & gx, RGBValue<Value> const & gy) const
00752     {
00753         result_type mag = (gx.red()*gx.red() +
00754                      gx.green()*gx.green() +
00755                      gx.blue()*gx.blue() +
00756                      gy.red()*gy.red() +
00757                      gy.green()*gy.green() +
00758                      gy.blue()*gy.blue()) / weight_;
00759 
00760         return (mag == zero_) ? one_ : one_ - VIGRA_CSTD::exp(-3.315 / mag / mag);
00761     }
00762     
00763     result_type weight_;
00764     result_type one_;
00765     result_type zero_;
00766 };
