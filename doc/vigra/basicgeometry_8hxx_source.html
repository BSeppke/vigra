00058 /** \brief Rotate an image by a multiple of 90 degrees or by an arbitrary angle.
00059 
00060     If you specify the angle as an integer which is a multiple of 90 degrees, rotateImage()
00061     just copies the pixels in the appropriate new order. It expects the destination image to 
00062     have the correct shape for the desired rotation. That is, when the rotation is a multiple
00063     of 180 degrees, source and destination must have the same shape, otherwise destination
00064     must have the transposed shape of the source.
00065     
00066     If you want to rotate by an arbitrary angle and around an arbitrary center point,
00067     you must specify the source image as a \ref vigra::SplineImageView, which is used for
00068     interpolation at the required subpixel positions. If no center point is provided, the image 
00069     center is used by default. The destination image must have the same size
00070     as the source SplineImageView.

00192           class DestIterator, class DestAccessor>
00193 void rotateImage(SrcIterator is, SrcIterator end, SrcAccessor as,
00194                            DestIterator id, DestAccessor ad, int rotation)
00195 {
00196     int x, y;
00197     int ws = end.x - is.x;
00198     int hs = end.y - is.y;
00199 
00200     vigra_precondition(rotation % 90 == 0, 
00201                 "rotateImage(): "
00202                 "This function rotates images only about multiples of 90 degree");
00203 
00204     rotation = rotation%360; 
00205     if (rotation < 0)
00206         rotation += 360;
00207     
00208     switch(rotation)
00209     {
00210         case 0:
00211             copyImage(is, end, as, id, ad);
00212             break;
00213         case 90: 
00214             is.x += (ws-1);
00215             for(x=0; x != ws; x++, is.x--, id.y++)
00216             {
00217                 typename SrcIterator::column_iterator cs = is.columnIterator();
00218                 typename DestIterator::row_iterator rd = id.rowIterator();
00219                 for(y=0; y != hs; y++, cs++, rd++)
00220                 {
00221                     ad.set(as(cs), rd);
00222                 }
00223         
00224             }
00225             break;
00226 
00227         case 180:
00228             end.x--;
00229             end.y--;
00230             for(x=0; x != ws; x++, end.x--, id.x++)
00231             {
00232                 typename SrcIterator::column_iterator cs = end.columnIterator();
00233                 typename DestIterator::column_iterator cd = id.columnIterator();
00234                 for(y=0; y != hs; y++, cs--, cd++)
00235                 {
00236                     ad.set(as(cs), cd);
00237                 }
00238         
00239             }
00240             break;
00241 
00242         case 270:  
00243             is.y += (hs-1);
00244             for(x=0; x != ws; x++, is.x++, id.y++)
00245             {
00246                 typename SrcIterator::column_iterator cs = is.columnIterator();
00247                 typename DestIterator::row_iterator rd = id.rowIterator();
00248                 for(y=0; y != hs; y++, cs--, rd++)
00249                 {
00250                     ad.set(as(cs), rd);

00471 /** \brief Transpose an image over the major or minor diagonal.
00472 
00473     The transposition direction refers to the axis, i.e.
00474     major transposition turns the upper right corner into the lower left one, 
00475     whereas minor transposition changes the upper left corner into the lower right one. 
00476     The directions are selected by the enum values
00477     <tt>vigra::major</tt> and <tt>vigra::minor</tt>. The two directions 
00478     can also be "or"ed together to perform both reflections simultaneously 
00479     (see example below) -- this is the same as a 180 degree rotation.
00480     (Caution: When doing multi-platform development, you should be
00481     aware that some <sys/types.h> define major/minor, too.  Do not omit
00482     the vigra namespace prefix.)

00484     Note that a similar effect can be chieved by MultiArrayView::transpose(). However,
00485     the latter can only transpose about the major diagonal, and it doesn't rearrange the data

00658 /*
00659 * Vergroessert eine Linie um einen Faktor. 
00660 * Ist z.B. der Faktor = 4 so werden in der
00661 * neuen Linie(Destination) jedes Pixel genau 4 mal 
00662 * vorkommen, also es findet auch keine Glaetung 
00663 * statt (NoInterpolation). Als Parameter sollen
00664 * der Anfangs-, der Enditerator und der Accessor
00665 * der Ausgangslinie (Source line), der Anfangsiterator
00666 * und Accessor der Ziellinie (destination line) und
00667 * anschliessend der Faktor um den die Linie (Zeile)
00668 * vergroessert bzw. verkleinert werden soll. 
00669 */
00670 template <class SrcIterator, class SrcAccessor, 
00671           class DestIterator, class DestAccessor>
00672 void resampleLine(SrcIterator src_iter, SrcIterator src_iter_end, SrcAccessor src_acc,
00673                   DestIterator dest_iter, DestAccessor dest_acc, double factor)
00674 {
00675     // The width of the src line.      
00676     int src_width = src_iter_end - src_iter;
00677  
00678     vigra_precondition(src_width > 0,
00679                        "resampleLine(): input image too small.");
00680     vigra_precondition(factor > 0.0,
00681                        "resampleLine(): factor must be positive.");
00682     
00683     if (factor >= 1.0)
00684     {
00685         int int_factor = (int)factor;
00686         double dx = factor - int_factor;
00687         double saver = dx;
00688         for ( ; src_iter != src_iter_end ; ++src_iter, saver += dx)
00689         {
00690             if (saver >= 1.0)
00691             {
00692                 saver = saver - (int)saver;
00693                 dest_acc.set(src_acc(src_iter), dest_iter);
00694                 ++dest_iter;
00695             }
00696             for(int i = 0 ; i < int_factor ; i++, ++dest_iter)
00697             {
00698                 dest_acc.set(src_acc(src_iter), dest_iter);
00699             }
00700         }
00701     }
00702     else
00703     {
00704         DestIterator dest_end = dest_iter + (int)VIGRA_CSTD::ceil(src_width*factor);  
00705         factor = 1.0/factor;
00706         int int_factor = (int)factor;
00707         double dx = factor - int_factor;
00708         double saver = dx;
00709         src_iter_end -= 1;
00710         for ( ; src_iter != src_iter_end && dest_iter != dest_end ; 
00711               ++dest_iter, src_iter += int_factor, saver += dx)
00712         {
00713             if (saver >= 1.0)
00714             {
00715                 saver = saver - (int)saver;
00716                 ++src_iter;
00717             }
00718             dest_acc.set(src_acc(src_iter), dest_iter);
00719         }
00720         if (dest_iter != dest_end)
00721         {
00722             dest_acc.set(src_acc(src_iter_end), dest_iter);
00723         }
00724     }
00725 }

00745     least 2x2. Destiniation pixels are directly copied from the appropriate
00746     source pixels. The size of the result image is the product of <tt>factor</tt>
00747     and the original size, where we round up if <tt>factor < 1.0</tt> and down otherwise.
00748     This size calculation is the main difference to the convention used in the similar 
00749     function \ref resizeImageNoInterpolation():

00859     int height_old = iend.y - is.y;

00861     //Bei Verkleinerung muss das dest-Bild ceiling(src*factor), da z.B.
00862     //aus 6x6 grossem Bild wird eins 18x18 grosses gemacht bei Vergroesserungsfaktor 3.1

00866     int width_new = sizeForResamplingFactor(width_old, xfactor);
