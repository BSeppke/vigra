00184 /** \brief Region Segmentation by means of Seeded Region Growing.
00185 
00186     This algorithm implements seeded region growing as described in
00187 
00188     R. Adams, L. Bischof: <em>"Seeded Region Growing"</em>, IEEE Trans. on Pattern
00189     Analysis and Maschine Intelligence, vol 16, no 6, 1994, and
00190 
00191     Ullrich K&ouml;the:
00192     <em><a href="http://hci.iwr.uni-heidelberg.de/people/ukoethe/papers/index.php#cite_primary_segmentation">Primary Image Segmentation</a></em>,
00193     in: G. Sagerer, S.
00194     Posch, F. Kummert (eds.): Mustererkennung 1995, Proc. 17. DAGM-Symposium,
00195     Springer 1995
00196 
00197     The seed image is a partly segmented image which contains uniquely
00198     labeled regions (the seeds) and unlabeled pixels (the candidates, label 0).
00199     The highest seed label found in the seed image is returned by the algorithm.
00200     
00201     Seed regions can be as large as you wish and as small as one pixel. If
00202     there are no candidates, the algorithm will simply copy the seed image
00203     into the output image. Otherwise it will aggregate the candidates into
00204     the existing regions so that a cost function is minimized. 
00205     Candidates are taken from the neighborhood of the already assigned pixels, 
00206     where the type of neighborhood is determined by parameter <tt>neighborhood</tt>
00207     which can take the values <tt>FourNeighborCode()</tt> (the default) 
00208     or <tt>EightNeighborCode()</tt>. The algorithm basically works as follows 
00209     (illustrated for 4-neighborhood, but 8-neighborhood works in the same way):
00210 
00211     <ol>
00212 
00213     <li> Find all candidate pixels that are 4-adjacent to a seed region.
00214     Calculate the cost for aggregating each candidate into its adjacent region
00215     and put the candidates into a priority queue.
00216 
00217     <li> While( priority queue is not empty and termination criterion is not fulfilled)
00218 
00219         <ol>
00220 
00221         <li> Take the candidate with least cost from the queue. If it has not
00222         already been merged, merge it with it's adjacent region.
00223 
00224         <li> Put all candidates that are 4-adjacent to the pixel just processed
00225         into the priority queue.
00226 
00227         </ol>
00228 
00229     </ol>
00230 
00231     <tt>SRGType</tt> can take the following values:
00232     
00233     <DL>
00234     <DT><tt>CompleteGrow</tt> <DD> produce a complete tesselation of the volume (default).
00235     <DT><tt>KeepContours</tt> <DD> keep a 1-voxel wide unlabeled contour between all regions.
00236     <DT><tt>StopAtThreshold</tt> <DD> stop when the boundary indicator values exceed the 
00237                              threshold given by parameter <tt>max_cost</tt>.
00238     <DT><tt>KeepContours | StopAtThreshold</tt> <DD> keep 1-voxel wide contour and stop at given <tt>max_cost</tt>.
00239     </DL>
00240 
00241     The cost is determined jointly by the source image and the
00242     region statistics functor. The source image contains feature values for each
00243     pixel which will be used by the region statistics functor to calculate and
00244     update statistics for each region and to calculate the cost for each
00245     candidate. The <TT>RegionStatisticsArray</TT> must be compatible to the
00246     \ref ArrayOfRegionStatistics functor and contains an <em> array</em> of
00247     statistics objects for each region. The indices must correspond to the
00248     labels of the seed regions. The statistics for the initial regions must have
00249     been calculated prior to calling <TT>seededRegionGrowing()</TT> (for example by
00250     means of \ref inspectTwoImagesIf()).
00251 
00252     For each candidate
00253     <TT>x</TT> that is adjacent to region <TT>i</TT>, the algorithm will call
00254     <TT>stats[i].cost(as(x))</TT> to get the cost (where <TT>x</TT> is a <TT>SrcIterator</TT>
00255     and <TT>as</TT> is
00256     the SrcAccessor). When a candidate has been merged with a region, the
00257     statistics are updated by calling <TT>stats[i].operator()(as(x))</TT>. Since
00258     the <TT>RegionStatisticsArray</TT> is passed by reference, this will overwrite
00259     the original statistics.
00260 
00261     If a candidate could be merged into more than one regions with identical
00262     cost, the algorithm will favour the nearest region. If <tt>StopAtThreshold</tt> is active, 
00263     and the cost of the current candidate at any point in the algorithm exceeds the optional 
00264     <tt>max_cost</tt> value (which defaults to <tt>NumericTraits<double>::max()</tt>), 
00265     region growing is aborted, and all voxels not yet assigned to a region remain unlabeled.
00266 
00267     In some cases, the cost only depends on the feature value of the current
00268     pixel. Then the update operation will simply be a no-op, and the <TT>cost()</TT>
00269     function returns its argument. This behavior is implemented by the
00270     \ref SeedRgDirectValueFunctor. With <tt>SRGType == KeepContours</tt>,
00271     this is equivalent to the watershed algorithm.

00415 template <class SrcIterator, class SrcAccessor,
00416           class SeedImageIterator, class SeedAccessor,
00417           class DestIterator, class DestAccessor,
00418           class RegionStatisticsArray, class Neighborhood>
00419 typename SeedAccessor::value_type
00420 seededRegionGrowing(SrcIterator srcul,
00421                     SrcIterator srclr, SrcAccessor as,
00422                     SeedImageIterator seedsul, SeedAccessor aseeds,
00423                     DestIterator destul, DestAccessor ad,
00424                     RegionStatisticsArray & stats,
00425                     SRGType srgType,
00426                     Neighborhood,
00427                     double max_cost)
00428 {
00429     int w = srclr.x - srcul.x;
00430     int h = srclr.y - srcul.y;
00431     int count = 0;
00432 
00433     SrcIterator isy = srcul, isx = srcul;  // iterators for the src image
00434 
00435     typedef typename SeedAccessor::value_type LabelType;
00436     typedef typename RegionStatisticsArray::value_type RegionStatistics;
00437     typedef typename RegionStatistics::cost_type CostType;
00438     typedef detail::SeedRgPixel<CostType> Pixel;
00439 
00440     typename Pixel::Allocator allocator;
00441 
00442     typedef std::priority_queue<Pixel *, std::vector<Pixel *>,
00443                                 typename Pixel::Compare>  SeedRgPixelHeap;
00444 
00445     // copy seed image in an image with border
00446     IImage regions(w+2, h+2);
00447     IImage::Iterator ir = regions.upperLeft() + Diff2D(1,1);
00448     IImage::Iterator iry, irx;
00449 
00450     initImageBorder(destImageRange(regions), 1, SRGWatershedLabel);
00451     copyImage(seedsul, seedsul+Diff2D(w,h), aseeds, ir, regions.accessor());
00452 
00453     // allocate and init memory for the results
00454 
00455     SeedRgPixelHeap pheap;
00456     int cneighbor, maxRegionLabel = 0;
00457     
00458     typedef typename Neighborhood::Direction Direction;
00459     int directionCount = Neighborhood::DirectionCount;
00460     
00461     Point2D pos(0,0);
00462     for(isy=srcul, iry=ir, pos.y=0; pos.y<h;
00463         ++pos.y, ++isy.y, ++iry.y)
00464     {
00465         for(isx=isy, irx=iry, pos.x=0; pos.x<w;
00466             ++pos.x, ++isx.x, ++irx.x)
00467         {
00468             if(*irx == 0)
00469             {
00470                 // find candidate pixels for growing and fill heap
00471                 for(int i=0; i<directionCount; i++)
00472                 {
00473                     // cneighbor = irx[dist[i]];
00474                     cneighbor = irx[Neighborhood::diff((Direction)i)];
00475                     if(cneighbor > 0)
00476                     {
00477                         CostType cost = stats[cneighbor].cost(as(isx));
00478 
00479                         Pixel * pixel =
00480                             allocator.create(pos, pos+Neighborhood::diff((Direction)i), cost, count++, cneighbor);
00481                         pheap.push(pixel);
00482                     }
00483                 }
00484             }
00485             else
00486             {
00487                 vigra_precondition((LabelType)*irx <= stats.maxRegionLabel(),
00488                     "seededRegionGrowing(): Largest label exceeds size of RegionStatisticsArray.");
00489                 if(maxRegionLabel < *irx)
00490                     maxRegionLabel = *irx;
00491             }
00492         }
00493     }
00494     
00495     // perform region growing
00496     while(pheap.size() != 0)
00497     {
00498         Pixel * pixel = pheap.top();
00499         pheap.pop();
00500 
00501         Point2D pos = pixel->location_;
00502         Point2D nearest = pixel->nearest_;
00503         int lab = pixel->label_;
00504         CostType cost = pixel->cost_;
00505 
00506         allocator.dismiss(pixel);
00507 
00508         if((srgType & StopAtThreshold) != 0 && cost > max_cost)
00509             break;
00510 
00511         irx = ir + pos;
00512         isx = srcul + pos;
00513 
00514         if(*irx) // already labelled region / watershed?
00515             continue;
00516 
00517         if((srgType & KeepContours) != 0)
00518         {
00519             for(int i=0; i<directionCount; i++)
00520             {
00521                 cneighbor = irx[Neighborhood::diff((Direction)i)];
00522                 if((cneighbor>0) && (cneighbor != lab))
00523                 {
00524                     lab = SRGWatershedLabel;
00525                     break;
00526                 }
00527             }
00528         }
00529 
00530         *irx = lab;
00531 
00532         if((srgType & KeepContours) == 0 || lab > 0)
00533         {
00534             // update statistics
00535             stats[*irx](as(isx));
00536 
00537             // search neighborhood
00538             // second pass: find new candidate pixels
00539             for(int i=0; i<directionCount; i++)
00540             {
00541                 if(irx[Neighborhood::diff((Direction)i)] == 0)
00542                 {
00543                     CostType cost = stats[lab].cost(as(isx, Neighborhood::diff((Direction)i)));
00544 
00545                     Pixel * new_pixel =
00546                         allocator.create(pos+Neighborhood::diff((Direction)i), nearest, cost, count++, lab);
00547                     pheap.push(new_pixel);
00548                 }
00549             }
00550         }
00551     }
00552     
00553     // free temporary memory
00554     while(pheap.size() != 0)
00555     {
00556         allocator.dismiss(pheap.top());
00557         pheap.pop();
00558     }
00559 
00560     // write result
00561     transformImage(ir, ir+Point2D(w,h), regions.accessor(), destul, ad,
00562                    detail::UnlabelWatersheds());
00563 
00564     return (LabelType)maxRegionLabel;
00565 }

00739 template <class SrcIterator, class SrcAccessor,
00740           class DestIterator, class DestAccessor,
00741           class RegionStatisticsArray, class Neighborhood>
00742 typename DestAccessor::value_type 
00743 fastSeededRegionGrowing(SrcIterator srcul, SrcIterator srclr, SrcAccessor as,
00744                         DestIterator destul, DestAccessor ad,
00745                         RegionStatisticsArray & stats,
00746                         SRGType srgType,
00747                         Neighborhood,
00748                         double max_cost,
00749                         std::ptrdiff_t bucket_count = 256)
00750 {
00751     typedef typename DestAccessor::value_type LabelType;
00752 
00753     vigra_precondition((srgType & KeepContours) == 0,
00754        "fastSeededRegionGrowing(): the turbo algorithm doesn't support 'KeepContours', sorry.");
00755     
00756     int w = srclr.x - srcul.x;
00757     int h = srclr.y - srcul.y;
00758 
00759     SrcIterator isy = srcul, isx = srcul;  // iterators for the src image
00760     DestIterator idy = destul, idx = destul;  // iterators for the dest image
00761 
00762     BucketQueue<Point2D, true> pqueue(bucket_count);
00763     LabelType maxRegionLabel = 0;
00764     
00765     Point2D pos(0,0);
00766     for(isy=srcul, idy = destul, pos.y=0; pos.y<h; ++pos.y, ++isy.y, ++idy.y)
00767     {
00768         for(isx=isy, idx=idy, pos.x=0; pos.x<w; ++pos.x, ++isx.x, ++idx.x)
00769         {
00770             LabelType label = ad(idx);
00771             if(label != 0)
00772             {
00773                 vigra_precondition(label <= stats.maxRegionLabel(),
00774                     "fastSeededRegionGrowing(): Largest label exceeds size of RegionStatisticsArray.");
00775 
00776                 if(maxRegionLabel < label)
00777                     maxRegionLabel = label;
00778                 
00779                 AtImageBorder atBorder = isAtImageBorder(pos.x, pos.y, w, h);
00780                 if(atBorder == NotAtBorder)
00781                 {
00782                     NeighborhoodCirculator<DestIterator, Neighborhood> c(idx), cend(c);
00783                     do
00784                     {
00785                         if(ad(c) == 0)
00786                         {
00787                             std::ptrdiff_t priority = (std::ptrdiff_t)stats[label].cost(as(isx));
00788                             pqueue.push(pos, priority);
00789                             break;
00790                         }
00791                     }
00792                     while(++c != cend);
00793                 }
00794                 else
00795                 {
00796                     RestrictedNeighborhoodCirculator<DestIterator, Neighborhood> 
00797                                                             c(idx, atBorder), cend(c);
00798                     do
00799                     {
00800                         if(ad(c) == 0)
00801                         {
00802                             std::ptrdiff_t priority = (std::ptrdiff_t)stats[label].cost(as(isx));
00803                             pqueue.push(pos, priority);
00804                             break;
00805                         }
00806                     }
00807                     while(++c != cend);
00808                 }
00809             }
00810         }
00811     }
00812     
00813     // perform region growing
00814     while(!pqueue.empty())
00815     {
00816         Point2D pos = pqueue.top();
00817         std::ptrdiff_t cost = pqueue.topPriority();
00818         pqueue.pop();
00819         
00820         if((srgType & StopAtThreshold) != 0 && cost > max_cost)
00821             break;
00822 
00823         idx = destul + pos;
00824         isx = srcul + pos;
00825         
00826         std::ptrdiff_t label = ad(idx);
00827 
00828         AtImageBorder atBorder = isAtImageBorder(pos.x, pos.y, w, h);
00829         if(atBorder == NotAtBorder)
00830         {
00831             NeighborhoodCirculator<DestIterator, Neighborhood> c(idx), cend(c);
00832             
00833             do
00834             {
00835                 std::ptrdiff_t nlabel = ad(c);
00836                 if(nlabel == 0)
00837                 {
00838                     ad.set(label, idx, c.diff());
00839                     std::ptrdiff_t priority = 
00840                            std::max((std::ptrdiff_t)stats[label].cost(as(isx, c.diff())), cost);
00841                     pqueue.push(pos+c.diff(), priority);
00842                 }
00843             }
00844             while(++c != cend);
00845         }
00846         else
00847         {
00848             RestrictedNeighborhoodCirculator<DestIterator, Neighborhood> 
00849                                                     c(idx, atBorder), cend(c);
00850             do
00851             {
00852                 std::ptrdiff_t nlabel = ad(c);
00853                 if(nlabel == 0)
00854                 {
00855                     ad.set(label, idx, c.diff());
00856                     std::ptrdiff_t priority = 
00857                            std::max((std::ptrdiff_t)stats[label].cost(as(isx, c.diff())), cost);
00858                     pqueue.push(pos+c.diff(), priority);
00859                 }
00860             }
00861             while(++c != cend);
00862         }
00863     }
00864     
00865     return maxRegionLabel;
00866 }
