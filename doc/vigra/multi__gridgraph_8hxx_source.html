00161 gridGraphEdgeCount(Shape const & shape, NeighborhoodType t, bool directed)
00162 {
00163     int res = 0;
00164     if(t == DirectNeighborhood)
00165     {
00166         for(unsigned int k=0; k<shape.size(); ++k)
00167             res += 2*prod(shape - Shape::unitVector(k));
00168     }
00169     else
00170     {
00171         res = prod(3*shape - Shape(2)) - prod(shape);
00172     }
00173     return directed
00174                ? res
00175                : res / 2;

00200         incrementOffsets[k].clear();
00201         edgeDescriptorOffsets[k].clear();
00202         indices[k].clear();
00203         backIndices[k].clear();
00204         
00205         for(unsigned int j=0; j < neighborOffsets.size(); ++j)
00206         {
00207             if(neighborExists[k][j])
00208             {
00209                 if(incrementOffsets[k].size() == 0)
00210                 {
00211                     incrementOffsets[k].push_back(neighborOffsets[j]);
00212                 }
00213                 else
00214                 {
00215                     incrementOffsets[k].push_back(neighborOffsets[j] - neighborOffsets[indices[k].back()]);

00223                 {
00224                     edgeDescriptorOffsets[k].push_back(EdgeDescriptor(neighborOffsets[j], neighborOffsets.size()-j-1, true));

00227                 {
00228                     edgeDescriptorOffsets[k].push_back(EdgeDescriptor(neighborOffsets[j] - neighborOffsets[indices[k].back()],

00230                 }
