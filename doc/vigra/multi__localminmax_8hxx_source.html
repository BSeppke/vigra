00054 template <class Graph, class T1Map, class T2Map, class Compare>
00055 unsigned int
00056 localMinMaxGraph(Graph const &g, 
00057                  T1Map const &src,
00058                  T2Map &dest,
00059                  typename property_traits<T2Map>::value_type marker,
00060                  typename property_traits<T1Map const>::value_type threshold,
00061                  Compare const &compare,
00062                  bool allowAtBorder = true)
00063 {
00064     typedef typename graph_traits<Graph>::vertex_iterator graph_scanner;
00065     typedef typename graph_traits<Graph>::adjacency_iterator neighbor_iterator;
00066 
00067     typedef typename property_traits<T1Map const>::value_type T1;
00068 
00069     graph_scanner node, srcend;
00070     neighbor_iterator arc, nbend;
00071 
00072     unsigned int count = 0;
00073     tie(node, srcend) = vertices(g);
00074     for (; node != srcend; ++node) 
00075     {
00076         const T1 current = get(src, *node);
00077 
00078         if (!compare(current, threshold))
00079             continue;
00080           
00081         if(!allowAtBorder && node.atBorder())
00082             continue;
00083         
00084         tie(arc, nbend) = adjacent_vertices(*node, g);
00085         for (;arc != nbend; ++arc) 
00086             if (!compare(current, get(src, *arc))) 
00087                 break;
00088                 
00089         if (arc == nbend)
00090         {
00091             put(dest, *node, marker);
00092             ++count;
00093         }
00094     }
00095     return count;
00096 }

00104 localMinMaxGraph(Graph const &g, 
00105                  T1Map const &src,

00107                  typename T2Map::value_type marker,
00108                  typename T1Map::value_type threshold,
00109                  Compare const &compare,
00110                  bool allowAtBorder = true)
00111 {
00112     typedef typename Graph::NodeIt    graph_scanner;
00113     typedef typename Graph::OutArcIt  neighbor_iterator;
00114 
00115     unsigned int count = 0;
00116     for (graph_scanner node(g); node != INVALID; ++node)

00119

00121             continue;

00123         if(!allowAtBorder && node.atBorder())
00124             continue;
00125

00140 template <class Graph, class T1Map, class T2Map, class Compare, class Equal>
00141 unsigned int
00142 extendedLocalMinMaxGraph(Graph const &g, 
00143                          T1Map const &src,
00144                          T2Map &dest,
00145                          typename T2Map::value_type marker,
00146                          typename T1Map::value_type threshold,
00147                          Compare const &compare,
00148                          Equal const &equal,
00149                          bool allowAtBorder = true)
00150 {
00151     typename Graph::template NodeMap<unsigned int> regions(g);
00152     
00153     int max_region_label = labelGraph(g, src, regions, equal);
00154     
00155     // assume that a region is a extremum until the opposite is proved
00156     std::vector<unsigned char> isExtremum(max_region_label+1, (unsigned char)1);
00157 
00158     typedef typename Graph::NodeIt    graph_scanner;
00159     typedef typename Graph::OutArcIt  neighbor_iterator;
00160 
00161     unsigned int count = max_region_label;
00162     for (graph_scanner node(g); node != INVALID; ++node) 
00163     {
00164         unsigned int label = regions[*node];
00165         
00166         if(!isExtremum[label])
00167             continue;
00168         
00169         typename T1Map::value_type current = src[*node];
00170 
00171         if (!compare(current, threshold) ||
00172             (!allowAtBorder && node.atBorder()))
00173         {
00174             isExtremum[label] = 0;
00175             --count;
00176             continue;
00177         }
00178         
00179         for (neighbor_iterator arc(g, node); arc != INVALID; ++arc) 
00180         {
00181             if (label != regions[g.target(*arc)] && compare(src[g.target(*arc)], current)) 
00182             {
00183                 isExtremum[label] = 0;
00184                 --count;
00185                 break;
00186             }
00187         }
00188     }
00189     for (graph_scanner node(g); node != INVALID; ++node) 
00190     {
00191         if(isExtremum[regions[*node]])
00192             dest[*node] = marker;
00193     }
00194     return count;
00195 }

00200                           class T2, class C2,
00201           class Compare,

00203 unsigned int
00204 localMinMax(MultiArrayView<N, T1, C1> const & src,
00205             MultiArrayView<N, T2, C2> dest,
00206             T1 threshold,
00207             Compare const & compare,
00208             EqualityFunctor const & equal,
00209             LocalMinmaxOptions const & options = LocalMinmaxOptions())
00210 {
00211     vigra_precondition(src.shape() == dest.shape(),
00212         "localMinMax(): shape mismatch between input and output.");
00213         
00214     NeighborhoodType neighborhood;
00215     
00216     if(options.neigh == 0 || options.neigh == 2*N)
00217         neighborhood = DirectNeighborhood;
00218     else if(options.neigh == 1 || options.neigh == MetaPow<3, N>::value - 1)
00219         neighborhood = IndirectNeighborhood;
00220     else
00221         vigra_precondition(false,
00222             "localMinMax(): option object specifies invalid neighborhood type.");
00223     
00224     T2 marker = (T2)options.marker;
00225     
00226     GridGraph<N, undirected_tag> graph(src.shape(), neighborhood);
00227     if(options.allow_plateaus)
00228         return lemon_graph::extendedLocalMinMaxGraph(graph, src, dest, marker, threshold, 
00229                                             compare, equal, options.allow_at_border);
00230     else
00231         return lemon_graph::localMinMaxGraph(graph, src, dest, marker, threshold, 
00232                                              compare, options.allow_at_border);
00233 }
