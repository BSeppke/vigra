00047 /** Number type for automatic differentiation.
00048 
00049     <a href="http://en.wikipedia.org/wiki/Automatic_differentiation">Automatic differentiation</a> 
00050     allows one to compute the value of a numeric expression and its gradient 
00051     with respect to the expression's arguments automatically and in one go. 
00052     To support this, one needs a special number type that holds a scalar value
00053     and the corresponding gradient vector of appropriate length. This is the
00054     purpose of hte template class <tt>DualVector<T, N></tt>, where <tt>T</tt> is the
00055     underlying numerical type (usually 'double'), and <tt>N</tt> denotes the 
00056     length of the gradient vector. 
00057     
00058     The standard arithmetic and algebraic functions are overloaded for 
00059     <tt>DualVector</tt> in order to implement the required arithmetic of 
00060     dual numbers. When you replace all arguments in a numeric expression
00061     with the appropriate <tt>DualVector</tt> instances, the result will be
00062     a <tt>DualVector</tt> that contains the result value and gradient of
00063     the expression, evaluated at the point defined by the input values.
00064     
00065     <b> Usage:</b>
00066     
00067     <b>\#include</b> <vigra/autodiff.hxx><br>
00068     Namespace: vigra::autodiff
00069 
00070     \code
00071     typedef DualVector<double, 2> N;  // for expressions with two arguments
00072     
00073     N x(1.0, 0);  // first argument of the expression
00074     N s(2.0, 1);  // second argument of the expression
00075     
00076     N y = exp(-0.5 * sq(x / s));
00077     
00078     std::cout << "Evaluated exp(- x^2 / (2 s^2)) at x=1 and s = 2:\n";
00079     std::cout << "result = " << y.value() <<", gradient = " << y.gradient() << "\n";
00080     \endcode
00081     Note that the second argument of the <tt>DualVector</tt> constructors specifies that 
00082     the derivative w.r.t 'x' shall be the element 0 of the gradient vector, and the
00083     derivative w.r.t. 's' shall be element 1.
00084 */
00085 template <class T, int N>
00086 class DualVector
00087 {
00088   public:
00089     typedef T                 value_type; ///< type of function values and gradient elements
00090     typedef TinyVector<T, N>  Gradient;   ///< type of the gradient vector
00091     
00092     T v;
00093     Gradient d;
00094     
00095         /** Zero initialization.
00096         */
00097     DualVector()
00098     : v(), d()
00099     {}
00100     
00101         /** Provide a value, but zero-initialize the gradient.
00102         */
00103     explicit DualVector(T const & val)
00104     : v(val), d()
00105     {}
00106     
00107         /** Initialize with given value and gradient.
00108         */
00109     DualVector(T const & val, Gradient const & grad)
00110     : v(val), d(grad)
00111     {}
00112     
00113         /** Shorthand for <tt>DualVector(val, Gradient(g0))</tt> when <tt>N == 1</tt>.
00114         
00115             Not to be used when <tt>N != 1</tt>.
00116         */
00117     DualVector(T const & val, T const & g0)
00118     : v(val), d(g0)
00119     {}
00120     
00121         /** Shorthand for <tt>DualVector(val, Gradient(g0, g1))</tt> when <tt>N == 2</tt>.
00122         
00123             Not to be used when <tt>N != 2</tt>.
00124         */
00125     DualVector(T const & val, T const & g0, T const & g1)
00126     : v(val), d(g0, g1)
00127     {}
00128     
00129         /** Initialize value to represent the argument number 'targetElement' in an
00130             expression. 
00131 
00132             The derivative of the expression w.r.t. this variable will be element 'targetElement'
00133             of the resulting gradient vector.
00134         */
00135     DualVector(T const & val, int targetElement)
00136     : v(val), d()
00137     {
00138         d[targetElement] = T(1.0);
00139     }
00140     
00141         /** Get current value.
00142         */
00143     T value() const
00144     {
00145         return v;
00146     }
00147     
00148         /** Get current gradient.
00149         */
00150     Gradient const & gradient() const
00151     {
00152         return d;
00153     }
00154

00156     {
00157         return *this;
00158     }
00159     
00160     DualVector operator-() const
00161     {
00162         return DualVector(-v, -d);
00163     }
00164     
00165     DualVector & operator+=(DualVector const & o)
00166     {
00167         d += o.d;
00168         v += o.v;
00169         return *this;
00170     }

00173     {
00174         v += o;

00176     }
00177     
00178     DualVector & operator-=(DualVector const & o)
00179     {

00182         return *this;
00183     }
00184

00186     {

00189     }
00190     
00191     DualVector & operator*=(DualVector const & o)
00192     {
00193         d = o.v * d + v * o.d;
00194         v *= o.v;
00195         return *this;
00196     }
00197     
00198     DualVector & operator*=(T const & o)
00199     {
00200         d *= o;

00202         return *this;
00203     }
00204     
00205     DualVector & operator/=(DualVector const & o)
00206     {
00207         d = (o.v * d - v * o.d) / sq(o.v);
00208         v /= o.v;
00209         return *this;
00210     }
00211     
00212     DualVector & operator/=(T const & o)
00213     {
00214         d /= o;
00215         v /= o;
00216         return *this;
00217     }
00218 };

00225 dualMatrix(TinyVector<T, N> const & v)
00226 {

00228     for(int k=0; k<N; ++k)
00229     {
