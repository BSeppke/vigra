00064                    KernelIterator ki, KernelAccessor ak,
00065                    Diff2D kul, Diff2D klr, BorderTreatmentMode border)

00067     vigra_precondition((border == BORDER_TREATMENT_CLIP    ||
00068                         border == BORDER_TREATMENT_AVOID   ||
00069                         border == BORDER_TREATMENT_REFLECT ||
00070                         border == BORDER_TREATMENT_REPEAT  ||
00071                         border == BORDER_TREATMENT_WRAP    ||
00072                         border == BORDER_TREATMENT_ZEROPAD),
00073                        "convolveImage():\n"
00074                        "  Border treatment must be one of follow treatments:\n"
00075                        "  - BORDER_TREATMENT_CLIP\n"
00076                        "  - BORDER_TREATMENT_AVOID\n"
00077                        "  - BORDER_TREATMENT_REFLECT\n"
00078                        "  - BORDER_TREATMENT_REPEAT\n"
00079                        "  - BORDER_TREATMENT_WRAP\n"
00080                        "  - BORDER_TREATMENT_ZEROPAD\n");

00082     vigra_precondition(kul.x <= 0 && kul.y <= 0,
00083                        "convolveImage(): coordinates of "
00084                        "kernel's upper left must be <= 0.");
00085     vigra_precondition(klr.x >= 0 && klr.y >= 0,
00086                        "convolveImage(): coordinates of "
00087                        "kernel's lower right must be >= 0.");
00088 
00089     // use traits to determine SumType as to prevent possible overflow
00090     typedef typename
00091         PromoteTraits<typename SrcAccessor::value_type,
00092                       typename KernelAccessor::value_type>::Promote SumType;
00093     typedef typename
00094         NumericTraits<typename KernelAccessor::value_type>::RealPromote KernelSumType;
00095     typedef typename DestAccessor::value_type DestType;
00096 
00097     // calculate width and height of the image
00098     int w = src_lr.x - src_ul.x;
00099     int h = src_lr.y - src_ul.y;
00100 
00101     // calculate width and height of the kernel
00102     int kernel_width  = klr.x - kul.x + 1;
00103     int kernel_height = klr.y - kul.y + 1;
00104 
00105     vigra_precondition(w >= std::max(klr.x, -kul.x) + 1 && h >= std::max(klr.y, -kul.y) + 1,
00106                        "convolveImage(): kernel larger than image.");
00107 
00108     KernelSumType norm = KernelSumType();
00109     if(border == BORDER_TREATMENT_CLIP)
00110     {
00111         // calculate the sum of the kernel elements for renormalization
00112         KernelIterator yk  = ki + klr;
00113 
00114         // determine sum within kernel (= norm)
00115         for(int y = 0; y < kernel_height; ++y, --yk.y)
00116         {
00117             KernelIterator xk  = yk;
00118             for(int x = 0; x < kernel_width; ++x, --xk.x)
00119             {
00120                 norm += ak(xk);
00121             }
00122         }
00123         vigra_precondition(norm != NumericTraits<KernelSumType>::zero(),
00124             "convolveImage(): Cannot use BORDER_TREATMENT_CLIP with a DC-free kernel");
00125     }
00126 
00127     DestIterator yd = dest_ul;
00128     SrcIterator ys = src_ul;
00129 
00130     // iterate over the interior part
00131     for(int y=0; y<h; ++y, ++ys.y, ++yd.y)
00132     {
00133         // create x iterators
00134         DestIterator xd(yd);
00135         SrcIterator xs(ys);
00136 
00137         for(int x=0; x < w; ++x, ++xs.x, ++xd.x)
00138         {
00139             // init the sum
00140             SumType sum = NumericTraits<SumType>::zero();
00141             KernelIterator ykernel  = ki + klr;
00142             
00143             if(x >= klr.x && y >= klr.y && x < w + kul.x && y < h + kul.y)
00144             {
00145                 // kernel is entirely inside the image
00146                 SrcIterator yys = xs - klr;
00147                 SrcIterator yyend = xs - kul;
00148 
00149                 for(; yys.y <= yyend.y; ++yys.y, --ykernel.y)
00150                 {
00151                     typename SrcIterator::row_iterator xxs = yys.rowIterator();

00153                     typename KernelIterator::row_iterator xkernel= ykernel.rowIterator();
00154 
00155                     for(; xxs < xxe; ++xxs, --xkernel)
00156                     {
00157                         sum += ak(xkernel) * src_acc(xxs);
00158                     }
00159                 }
00160             }
00161             else if(border == BORDER_TREATMENT_REPEAT)
00162             {
00163                 Diff2D diff;
00164                 for(int yk = klr.y; yk >= kul.y; --yk, --ykernel.y)
00165                 {
00166                     diff.y = std::min(std::max(y - yk, 0), h-1);
00167                     typename KernelIterator::row_iterator xkernel  = ykernel.rowIterator();
00168 
00169                     for(int xk = klr.x; xk >= kul.x; --xk, --xkernel)
00170                     {
00171                         diff.x = std::min(std::max(x - xk, 0), w-1);
00172                         sum += ak(xkernel) * src_acc(src_ul, diff);
00173                     }
00174                 }
00175             }
00176             else if(border == BORDER_TREATMENT_REFLECT)
00177             {
00178                 Diff2D diff;
00179                 for(int yk = klr.y; yk >= kul.y; --yk , --ykernel.y)
00180                 {
00181                     diff.y = abs(y - yk);
00182                     if(diff.y >= h)
00183                         diff.y = 2*h - 2 - diff.y;
00184                     typename KernelIterator::row_iterator xkernel  = ykernel.rowIterator();
00185 
00186                     for(int xk = klr.x; xk >= kul.x; --xk, --xkernel)
00187                     {
00188                         diff.x = abs(x - xk);
00189                         if(diff.x >= w)
00190                             diff.x = 2*w - 2 - diff.x;
00191                         sum += ak(xkernel) * src_acc(src_ul, diff);
00192                     }
00193                 }
00194             }
00195             else if(border == BORDER_TREATMENT_WRAP)
00196             {
00197                 Diff2D diff;
00198                 for(int yk = klr.y; yk >= kul.y; --yk, --ykernel.y)
00199                 {
00200                     diff.y = (y - yk + h) % h;
00201                     typename KernelIterator::row_iterator xkernel  = ykernel.rowIterator();
00202 
00203                     for(int xk = klr.x; xk >= kul.x; --xk, --xkernel)
00204                     {
00205                         diff.x = (x - xk + w) % w;
00206                         sum += ak(xkernel) * src_acc(src_ul, diff);
00207                     }
00208                 }
00209             }
00210             else if(border == BORDER_TREATMENT_CLIP)
00211             {
00212                 KernelSumType ksum = NumericTraits<KernelSumType>::zero();
00213                 Diff2D diff;
00214                 for(int yk = klr.y; yk >= kul.y; --yk, --ykernel.y)
00215                 {
00216                     diff.y = y - yk;
00217                     if(diff.y < 0 || diff.y >= h)
00218                         continue;
00219                     typename KernelIterator::row_iterator xkernel  = ykernel.rowIterator();
00220 
00221                     for(int xk = klr.x; xk >= kul.x; --xk, --xkernel)
00222                     {
00223                         diff.x = x - xk;
00224                         if(diff.x < 0 || diff.x >= w)
00225                             continue;
00226                         ksum += ak(xkernel);
00227                         sum += ak(xkernel) * src_acc(src_ul, diff);
00228                     }
00229                 }
00230                 
00231                 sum *= norm / ksum;
00232             }
00233             else if(border == BORDER_TREATMENT_ZEROPAD)
00234             {
00235                 Diff2D diff;
00236                 for(int yk = klr.y; yk >= kul.y; --yk, --ykernel.y)
00237                 {
00238                     diff.y = y - yk;
00239                     if(diff.y < 0 || diff.y >= h)
00240                         continue;
00241                     typename KernelIterator::row_iterator xkernel  = ykernel.rowIterator();
00242 
00243                     for(int xk = klr.x; xk >= kul.x; --xk, --xkernel)
00244                     {
00245                         diff.x = x - xk;
00246                         if(diff.x < 0 || diff.x >= w)
00247                             continue;
00248                         sum += ak(xkernel) * src_acc(src_ul, diff);
00249                     }
00250                 }
00251             }
00252             else if(border == BORDER_TREATMENT_AVOID)
00253             {
00254                 continue;
00255             }
00256 
00257             // store convolution result in destination pixel
00258             dest_acc.set(detail::RequiresExplicitCast<DestType>::cast(sum), xd);
00259         }
00260     }
00261 }

00293 /** \brief Performs a 2-dimensional normalized convolution, i.e. convolution with a mask image.
00294 
00295     This functions computes
00296     <a href ="http://homepages.inf.ed.ac.uk/rbf/CVonline/LOCAL_COPIES/PIRODDI1/NormConv/NormConv.html">normalized
00297     convolution</a> as defined in
00298     Knutsson, H. and Westin, C-F.: <i>Normalized and differential convolution:
00299     Methods for Interpolation and Filtering of incomplete and uncertain data</i>.
00300     Proc. of the IEEE Conf. on Computer Vision and Pattern Recognition, 1993, 515-523.
00301 
00302     The mask image must be binary and encodes which pixels of the original image
00303     are valid. It is used as follows:
00304     Only pixel under the mask are used in the calculations. Whenever a part of the
00305     kernel lies outside the mask, it is ignored, and the kernel is renormalized to its
00306     original norm (analogous to the CLIP \ref BorderTreatmentMode). Thus, a useful convolution
00307     result is computed whenever <i>at least one valid pixel is within the current window</i>
00308     Thus, destination pixels not under the mask still receive a value if they are <i>near</i>
00309     the mask. Therefore, this algorithm is useful as an interpolator of sparse input data.
00310     If you are only interested in the destination values under the mask, you can perform
00311     a subsequent \ref copyImageIf().
00312 
00313     The KernelIterator must point to the center of the kernel, and
00314     the kernel's size is given by its upper left (x and y of distance <= 0) and
00315     lower right (distance >= 0) corners. The image must always be larger than the
00316     kernel. At those positions where the kernel does not completely fit
00317     into the image, the specified \ref BorderTreatmentMode is
00318     applied. Only BORDER_TREATMENT_CLIP and BORDER_TREATMENT_AVOID are currently
00319     supported.
00320 
00321     The images's pixel type (SrcAccessor::value_type) must be a
00322     linear space over the kernel's value_type (KernelAccessor::value_type),
00323     i.e. addition of source values, multiplication with kernel values,
00324     and NumericTraits must be defined.
00325     The kernel's value_type must be an algebraic field,
00326     i.e. the arithmetic operations (+, -, *, /) and NumericTraits must
00327     be defined.

00508

00522         MaskIterator xm(ym);
00523

00551                 typename KernelIterator::row_iterator xk  = yk.rowIterator();
00552 
00553                 for(xx=0; xxs < xxend; ++xxs, --xk, ++xxm)
00554                 {
00555                     if(!am(xxm)) continue;
00556 
00557                     if(first)
00558                     {
00559                         sum = detail::RequiresExplicitCast<SumType>::cast(ak(xk) * src_acc(xxs));
00560                         ksum = ak(xk);
00561                         first = false;
00562                     }
00563                     else
00564                     {
00565                         sum = detail::RequiresExplicitCast<SumType>::cast(sum + ak(xk) * src_acc(xxs));
00566                         ksum += ak(xk);
00567                     }
00568                 }
00569             }
00570             // store average in destination pixel
00571             if(ksum != NumericTraits<KSumType>::zero())
00572             {
00573                 dest_acc.set(DestTraits::fromRealPromote(
00574                              detail::RequiresExplicitCast<SumType>::cast((norm / ksum) * sum)), xd);
00575             }
00576         }
00577     }
00578 }

00717 /** \brief Generic 2 dimensional convolution kernel.
00718

00720 
00721     Convolution functions access the kernel via an ImageIterator
00722     which they get by calling \ref center(). This iterator
00723     points to the center of the kernel. The kernel's size is given by its upperLeft()
00724     (upperLeft().x <= 0, upperLeft().y <= 0)
00725     and lowerRight() (lowerRight().x >= 0, lowerRight().y >= 0) methods.
00726     The desired border treatment mode is returned by borderTreatment().
00727

00729     properties. The requirements for the kernel's value_type depend
00730     on the init function used. At least NumericTraits must be defined.

01033             <TT>NumericTraits<value_type>::one() / (number of non-zero kernel values)</TT>.

01052     {
01053         vigra_precondition(radius > 0,
01054                            "Kernel2D::initDisk(): radius must be > 0.");
01055 
01056         left_ = Point2D(-radius, -radius);
01057         right_ = Point2D(radius, radius);
01058         int w = right_.x - left_.x + 1;
01059         int h = right_.y - left_.y + 1;
01060         kernel_.resize(w, h);
