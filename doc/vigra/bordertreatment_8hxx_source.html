00043 /** \page BorderTreatmentMode BorderTreatmentMode
00044 
00045     Choose between different border treatment modes. In the convolution 
00046     algorithms, these modes apply to 
00047     all image pixels where the kernel does not completely fit inside 
00048     the image.
00049     
00050     <b>\#include</b> <vigra/bordertreatment.hxx><br>
00051     Namespace: vigra
00052     
00053     \code
00054     enum BorderTreatmentMode 
00055     {
00056           // do not operate on a pixel where the kernel does 
00057           // not fit in the image
00058        BORDER_TREATMENT_AVOID, 
00059 
00060           // clip kernel at image border (this is only useful if the
00061           //  kernel is >= 0 everywhere)
00062        BORDER_TREATMENT_CLIP, 
00063 
00064           // repeat the nearest valid pixel
00065        BORDER_TREATMENT_REPEAT,
00066 
00067           // reflect image at last row/column 
00068        BORDER_TREATMENT_REFLECT, 
00069 
00070           // wrap image around (periodic boundary conditions)
00071        BORDER_TREATMENT_WRAP
00072 
00073           // assume that all outside points have value zero
00074        BORDER_TREATMENT_ZEROPAD
00075     };
00076     \endcode
00077 */
