00055 unsigned int watershedLabeling(SrcIterator upperlefts,
00056                         SrcIterator lowerrights, SrcAccessor sa,
00057                         DestIterator upperleftd, DestAccessor da,
00058                         Neighborhood)
00059 {
00060     typedef typename DestAccessor::value_type LabelType;
00061 
00062     int w = lowerrights.x - upperlefts.x;
00063     int h = lowerrights.y - upperlefts.y;
00064     int x,y;
00065 
00066     SrcIterator ys(upperlefts);
00067     SrcIterator xs(ys);
00068     DestIterator yd(upperleftd);
00069     DestIterator xd(yd);
00070 
00071     // temporary image to store region labels
00072     detail::UnionFindArray<LabelType> labels;
00073 
00074     // initialize the neighborhood circulators
00075     NeighborOffsetCirculator<Neighborhood> ncstart(Neighborhood::CausalFirst);
00076     NeighborOffsetCirculator<Neighborhood> ncstartBorder(Neighborhood::North);
00077     NeighborOffsetCirculator<Neighborhood> ncend(Neighborhood::CausalLast);
00078     ++ncend;
00079     NeighborOffsetCirculator<Neighborhood> ncendBorder(Neighborhood::North);
00080     ++ncendBorder;
00081 
00082     // pass 1: scan image from upper left to lower right
00083     // to find connected components
00084 
00085     // Each component will be represented by a tree of pixels. Each
00086     // pixel contains the scan order address of its parent in the
00087     // tree.  In order for pass 2 to work correctly, the parent must
00088     // always have a smaller scan order address than the child.
00089     // Therefore, we can merge trees only at their roots, because the
00090     // root of the combined tree must have the smallest scan order
00091     // address among all the tree's pixels/ nodes.  The root of each
00092     // tree is distinguished by pointing to itself (it contains its
00093     // own scan order address). This condition is enforced whenever a
00094     // new region is found or two regions are merged
00095     da.set(labels.finalizeLabel(labels.nextFreeLabel()), xd);
00096 
00097     ++xs.x;
00098     ++xd.x;
00099     for(x = 1; x != w; ++x, ++xs.x, ++xd.x)
00100     {
00101         if((sa(xs) & Neighborhood::directionBit(Neighborhood::West)) ||
00102            (sa(xs, Neighborhood::west()) & Neighborhood::directionBit(Neighborhood::East)))
00103         {
00104             da.set(da(xd, Neighborhood::west()), xd);
00105         }
00106         else
00107         {
00108             da.set(labels.finalizeLabel(labels.nextFreeLabel()), xd);
00109         }
00110     }
00111 
00112     ++ys.y;
00113     ++yd.y;
00114     for(y = 1; y != h; ++y, ++ys.y, ++yd.y)
00115     {
00116         xs = ys;
00117         xd = yd;
00118 
00119         for(x = 0; x != w; ++x, ++xs.x, ++xd.x)
00120         {
00121             NeighborOffsetCirculator<Neighborhood> nc(x == w-1
00122                                                         ? ncstartBorder
00123                                                         : ncstart);
00124             NeighborOffsetCirculator<Neighborhood> nce(x == 0
00125                                                          ? ncendBorder
00126                                                          : ncend);
00127             LabelType currentLabel = labels.nextFreeLabel();
00128             for(; nc != nce; ++nc)
00129             {
00130                 if((sa(xs) & nc.directionBit()) || (sa(xs, *nc) & nc.oppositeDirectionBit()))
00131                 {
00132                     currentLabel = labels.makeUnion(da(xd,*nc), currentLabel);
00133                 }
00134             }
00135             da.set(labels.finalizeLabel(currentLabel), xd);
00136         }
00137     }
00138 
00139     unsigned int count = labels.makeContiguous();
00140 
00141     // pass 2: assign one label to each region (tree)
00142     // so that labels form a consecutive sequence 1, 2, ...
00143     yd = upperleftd;
00144     for(y=0; y != h; ++y, ++yd.y)
00145     {
00146         DestIterator xd(yd);
00147         for(x = 0; x != w; ++x, ++xd.x)
00148         {
00149             da.set(labels[da(xd)], xd);
00150         }
00151     }
00152     return count;
00153 }

00417 /** \brief Generate seeds for watershed computation and seeded region growing.
00418 
00419     The source image is a boundary indicator such as the gradient magnitude
00420     or the trace of the \ref boundaryTensor(). Seeds are generally generated
00421     at locations where the boundaryness (i.e. the likelihood of the point being on the
00422     boundary) is very small. In particular, seeds can be placed by either
00423     looking for local minima (possibly including minimal plateaus) of the boundaryness,
00424     of by looking at level sets (i.e. regions where the boundaryness is below a threshold).
00425     Both methods can also be combined, so that only minima below a threshold are returned.
00426     The particular seeding strategy is specified by the <tt>options</tt> object 
00427     (see \ref SeedOptions).
00428     
00429     The pixel type of the input image must be <tt>LessThanComparable</tt>.
00430     The pixel type of the output image must be large enough to hold the labels for all seeds.
00431     (typically, you will use <tt>UInt32</tt>). The function will label seeds by consecutive integers
00432     (starting from 1) and returns the largest label it used.
00433     
00434     Pass \ref vigra::NeighborhoodType "IndirectNeighborhood" or \ref vigra::NeighborhoodType "DirectNeighborhood" 
00435     (first form of the function) 
00436     or \ref vigra::EightNeighborCode or \ref vigra::FourNeighborCode (second and third forms) to determine the 
00437     neighborhood where pixel values are compared. 
00438 
00439     <b> Declarations:</b>

00493 template <class SrcIterator, class SrcAccessor,
00494           class DestIterator, class DestAccessor,
00495           class Neighborhood>
00496 unsigned int
00497 generateWatershedSeeds(SrcIterator upperlefts, SrcIterator lowerrights, SrcAccessor sa,
00498                        DestIterator upperleftd, DestAccessor da, 
00499                        Neighborhood neighborhood,
00500                        SeedOptions const & options = SeedOptions())
00501 {
00502     using namespace functor;
00503     typedef typename SrcAccessor::value_type SrcType;
00504     
00505     vigra_precondition(options.mini != SeedOptions::LevelSets || 
00506                        options.thresholdIsValid<SrcType>(),
00507         "generateWatershedSeeds(): SeedOptions.levelSets() must be specified with threshold.");
00508     
00509     Diff2D shape = lowerrights - upperlefts;
00510     BImage seeds(shape);
00511     
00512     if(options.mini == SeedOptions::LevelSets)
00513     {
00514         transformImage(srcIterRange(upperlefts, lowerrights, sa),
00515                        destImage(seeds),
00516                        ifThenElse(Arg1() <= Param(options.thresh), Param(1), Param(0)));
00517     }
00518     else
00519     {
00520         LocalMinmaxOptions lm_options;
00521         lm_options.neighborhood(Neighborhood::DirectionCount)
00522                   .markWith(1.0)
00523                   .allowAtBorder()
00524                   .allowPlateaus(options.mini == SeedOptions::ExtendedMinima);
00525         if(options.thresholdIsValid<SrcType>())
00526             lm_options.threshold(options.thresh);
00527             
00528         localMinima(srcIterRange(upperlefts, lowerrights, sa), destImage(seeds),
00529                     lm_options);
00530     }
00531     
00532     return labelImageWithBackground(srcImageRange(seeds), destIter(upperleftd, da), 
00533                                     Neighborhood::DirectionCount == 8, 0);
00534 }

01094 /** \brief Region segmentation by means of a flooding-based watershed algorithm.
01095 
01096     Note: This function is largely obsolete, \ref watershedsMultiArray() should be
01097     preferred unless top speed is required.
01098     
01099     This function implements variants of the watershed algorithm
01100     described in
01101 
01102     L. Vincent and P. Soille: <em>"Watersheds in digital spaces: An efficient algorithm
01103     based on immersion simulations"</em>, IEEE Trans. Patt. Analysis Mach. Intell. 13(6):583-598, 1991
01104 
01105     The source image is a boundary indicator such as the gaussianGradientMagnitude()
01106     or the trace of the \ref boundaryTensor(), and the destination is a label image
01107     designating membership of each point in one of the regions. Plateaus in the boundary
01108     indicator (i.e. regions of constant gray value) are handled via a Euclidean distance
01109     transform by default.
01110     
01111     By default, the destination image is assumed to hold seeds for a seeded watershed 
01112     transform. Seeds may, for example, be created by means of generateWatershedSeeds(). 
01113     Note that the seeds will be overridden with the final watershed segmentation.
01114     
01115     Alternatively, you may provide \ref SeedOptions in order to instruct 
01116     watershedsRegionGrowing() to generate its own seeds (it will call generateWatershedSeeds()
01117     internally). In that case, the destination image should be zero-initialized.
01118     
01119     You can specify the neighborhood system to be used by passing \ref FourNeighborCode 
01120     or \ref EightNeighborCode (default).
01121     
01122     Further options to be specified via \ref WatershedOptions are:
01123     
01124     <ul>
01125     <li> Whether to keep a 1-pixel-wide contour (with label 0) between regions or 
01126          perform complete grow (i.e. all pixels are assigned to a region).
01127     <li> Whether to stop growing when the boundaryness exceeds a threshold (remaining
01128          pixels keep label 0).
01129     <li> Whether to use a faster, but less powerful algorithm ("turbo algorithm"). It
01130          is faster because it orders pixels by means of a \ref BucketQueue (therefore,
01131          the boundary indicator must contain integers in the range 
01132          <tt>[0, ..., bucket_count-1]</tt>, where <tt>bucket_count</tt> is specified in
01133          the options object), it only supports complete growing (no contour between regions
01134          is possible), and it handles plateaus in a simplistic way. It also saves some
01135          memory because it allocates less temporary storage.
01136     <li> Whether one region (label) is to be preferred or discouraged by biasing its cost 
01137          with a given factor (smaller than 1 for preference, larger than 1 for discouragement).
01138     </ul>
01139

01141     \ref watershedsUnionFind().
