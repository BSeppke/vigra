00056 /** \brief Calculate the tensor (outer) product of a 2D vector with itself.
00057 
00058     This function is useful to transform vector images into a tensor representation 
00059     that can be used as input to tensor based processing and analysis functions
00060     (e.g. tensor smoothing). The input pixel type must be vectors of length 2, whereas
00061     the output must contain vectors of length 3 which will represent the tensor components
00062     in the order t11, t12 (== t21 due to symmetry), t22.

00065     the second tensor component (t12) can be negated by setting <tt>negateComponent2 = false</tt>.

00134 template <class SrcIterator, class SrcAccessor,
00135           class DestIterator, class DestAccessor>
00136 void vectorToTensor(SrcIterator sul, SrcIterator slr, SrcAccessor src,
00137                     DestIterator dul, DestAccessor dest,
00138                     bool negateComponent2)
00139 {
00140     vigra_precondition(src.size(sul) == 2,
00141                        "vectorToTensor(): input image must have 2 bands.");
00142     vigra_precondition(dest.size(dul) == 3,
00143                        "vectorToTensor(): output image must have 3 bands.");
00144 
00145     int w = slr.x - sul.x;
00146     int h = slr.y - sul.y;
00147 
00148     for(int y=0; y<h; ++y, ++sul.y, ++dul.y)
00149     {
00150         typename SrcIterator::row_iterator s = sul.rowIterator();
00151         typename SrcIterator::row_iterator send = s + w;
00152         typename DestIterator::row_iterator d = dul.rowIterator();
00153         if(negateComponent2)
00154         {
00155             for(; s < send; ++s, ++d)
00156             {
00157                 dest.setComponent(sq(src.getComponent(s, 0)), d, 0);
00158                 dest.setComponent(-src.getComponent(s, 0)*src.getComponent(s, 1), d, 1);
00159                                // ^ negative sign to turn left-handed into right-handed coordinates
00160                 dest.setComponent(sq(src.getComponent(s, 1)), d, 2);
00161             }
00162         }
00163         else
00164         {
00165             for(; s < send; ++s, ++d)
00166             {
00167                 dest.setComponent(sq(src.getComponent(s, 0)), d, 0);
00168                 dest.setComponent(src.getComponent(s, 0)*src.getComponent(s, 1), d, 1);
00169                 dest.setComponent(sq(src.getComponent(s, 1)), d, 2);
00170             }
00171         }
00172     }
00173 }

00287 template <class SrcIterator, class SrcAccessor,
00288           class DestIterator, class DestAccessor>
00289 void tensorEigenRepresentation(SrcIterator sul, SrcIterator slr, SrcAccessor src,
00290                                DestIterator dul, DestAccessor dest)
00291 {
00292     vigra_precondition(src.size(sul) == 3,
00293                        "tensorEigenRepresentation(): input image must have 3 bands.");
00294     vigra_precondition(dest.size(dul) == 3,
00295                        "tensorEigenRepresentation(): output image must have 3 bands.");
00296 
00297     int w = slr.x - sul.x;
00298     int h = slr.y - sul.y;
00299 
00300     for(int y=0; y<h; ++y, ++sul.y, ++dul.y)
00301     {
00302         typename SrcIterator::row_iterator s = sul.rowIterator();
00303         typename SrcIterator::row_iterator send = s + w;
00304         typename DestIterator::row_iterator d = dul.rowIterator();
00305         for(; s < send; ++s, ++d)
00306         {
00307             typedef typename 
00308                NumericTraits<typename SrcAccessor::component_type>::RealPromote TmpType;
00309             TmpType d1 = src.getComponent(s,0) + src.getComponent(s,2);
00310             TmpType d2 = src.getComponent(s,0) - src.getComponent(s,2);
00311             TmpType d3 = TmpType(2.0) * src.getComponent(s,1);
00312             TmpType d4 = (TmpType)hypot(d2, d3);
00313             
00314             dest.setComponent(0.5 * (d1 + d4), d, 0); // large EV
00315             dest.setComponent(0.5 * (d1 - d4), d, 1); // small EV
00316             if(d2==0.0 && d3==0.0)
00317             {
00318                 dest.setComponent(0, d, 2); // orientation
00319             }
00320             else
00321             {
00322                 dest.setComponent(0.5 * VIGRA_CSTD::atan2(d3, d2), d, 2); // orientation
00323             }
00324         }
00325     }
00326 }

00542 template <class SrcIterator, class SrcAccessor,
00543           class DestIterator1, class DestAccessor1,
00544           class DestIterator2, class DestAccessor2>
00545 void tensorToEdgeCorner(SrcIterator sul, SrcIterator slr, SrcAccessor src,
00546                         DestIterator1 edgeul, DestAccessor1 edge,
00547                         DestIterator2 cornerul, DestAccessor2 corner)
00548 {
00549     vigra_precondition(src.size(sul) == 3,
00550                        "tensorToEdgeCorner(): input image must have 3 bands.");
00551     vigra_precondition(edge.size(edgeul) == 2,
00552                        "tensorToEdgeCorner(): edge image must have 2 bands.");
00553 
00554     int w = slr.x - sul.x;
00555     int h = slr.y - sul.y;
00556 
00557     for(int y=0; y<h; ++y, ++sul.y, ++edgeul.y, ++cornerul.y)
00558     {
00559         typename SrcIterator::row_iterator s = sul.rowIterator();
00560         typename SrcIterator::row_iterator send = s + w;
00561         typename DestIterator1::row_iterator e = edgeul.rowIterator();
00562         typename DestIterator2::row_iterator c = cornerul.rowIterator();
00563         for(; s < send; ++s, ++e, ++c)
00564         {
00565             typedef typename 
00566                NumericTraits<typename SrcAccessor::component_type>::RealPromote TmpType;
00567             TmpType d1 = src.getComponent(s,0) + src.getComponent(s,2);
00568             TmpType d2 = src.getComponent(s,0) - src.getComponent(s,2);
00569             TmpType d3 = 2.0 * src.getComponent(s,1);
00570             TmpType d4 = (TmpType)hypot(d2, d3);
00571             
00572             edge.setComponent(d4, e, 0); // edgeness = difference of EVs
00573             if(d2 == 0.0 && d3 == 0.0)
00574             {
00575                 edge.setComponent(0.0, e, 1); // orientation
00576             }
00577             else
00578             {
00579                 edge.setComponent(0.5 * VIGRA_CSTD::atan2(d3, d2), e, 1); // orientation
00580             }
00581             corner.set(d1 - d4, c); // cornerness = 2 * small EV
00582         }
00583     }
00584 }
