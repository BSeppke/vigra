00061 /** \brief Create homogeneous matrix that maps corresponding points onto each other.
00062  
00063     For use with \ref affineWarpImage(). When only two corresponding points are given,
00064     the matrix will only represent a similarity transform 
00065     (translation, rotation, and uniform scaling). When only one point pair is given,
00066     the result will be a pure translation.
00067 */
00068 template <class SrcIterator, class DestIterator>
00069 linalg::TemporaryMatrix<double> 
00070 affineMatrix2DFromCorrespondingPoints(SrcIterator s, SrcIterator send, DestIterator d)
00071 {
00072     int size = send - s;
00073     
00074     linalg::TemporaryMatrix<double> ret(identityMatrix<double>(3));
00075 
00076     if(size == 1)
00077     {
00078         ret(0,2) = (*d)[0] - (*s)[0];
00079         ret(1,2) = (*d)[1] - (*s)[1];
00080     }
00081     else if(size == 2)
00082     {
00083         Matrix<double> m(4,4), r(4,1), so(4,1);
00084         
00085         for(int k=0; k<size; ++k, ++s, ++d)
00086         {
00087             m(2*k,0) = (*s)[0];
00088             m(2*k,1) = -(*s)[1];
00089             m(2*k,2) = 1.0;
00090             m(2*k,3) = 0.0;
00091             r(2*k,0) = (*d)[0];
00092             
00093             m(2*k+1,0) = (*s)[1];
00094             m(2*k+1,1) = (*s)[0];
00095             m(2*k+1,2) = 0.0;
00096             m(2*k+1,3) = 1.0;
00097             r(2*k+1,0) = (*d)[1];
00098         }
00099     
00100         if(!linearSolve(m, r, so))
00101             vigra_fail("affineMatrix2DFromCorrespondingPoints(): singular solution matrix.");
00102         
00103         ret(0,0) = so(0,0);
00104         ret(1,1) = so(0,0);
00105         ret(0,1) = -so(1,0);
00106         ret(1,0) = so(1,0);
00107         ret(0,2) = so(2,0);
00108         ret(1,2) = so(3,0);
00109     }
00110     else if(size >= 3)
00111     {
00112         Matrix<double> m(3,3),  rx(3,1), sx(3,1), ry(3,1), sy(3,1), c(3,1);
00113         c(2,0) = 1.0;
00114         for(int k=0; k<size; ++k, ++s, ++d)
00115         {
00116             c(0,0) = (*s)[0];
00117             c(1,0) = (*s)[1];
00118             
00119             m  += outer(c);
00120             rx += (*d)[0]*c;
00121             ry += (*d)[1]*c;
00122         }
00123     
00124         if(!linearSolve(m, rx, sx) || !linearSolve(m, ry, sy))
00125             vigra_fail("affineMatrix2DFromCorrespondingPoints(): singular solution matrix.");
00126         
00127         ret(0,0) = sx(0,0);
00128         ret(0,1) = sx(1,0);
00129         ret(0,2) = sx(2,0);
00130         ret(1,0) = sy(0,0);
00131         ret(1,1) = sy(1,0);
00132         ret(1,2) = sy(2,0);
00133     }
00134 
00135     return ret;
00136 }

00204 struct TranslationEstimationFunctor
00205 {
00206     template <class SplineImage, class Image>
00207     void operator()(SplineImage const & src, Image const & dest, Matrix<double> & matrix) const
00208     {
00209         int w = dest.width();
00210         int h = dest.height();
00211         
00212         Matrix<double> grad(2,1), m(2,2), r(2,1), s(2,1);
00213         double dx = matrix(0,0), dy = matrix(1,0);
00214         
00215         for(int y = 0; y < h; ++y)
00216         {
00217             double sx = matrix(0,1)*y + matrix(0,2);
00218             double sy = matrix(1,1)*y + matrix(1,2);
00219             for(int x = 0; x < w; ++x, sx += dx, sy += dy)
00220             {
00221                 if(!src.isInside(sx, sy))
00222                     continue;
00223                   
00224                 grad(0,0) = src.dx(sx, sy);
00225                 grad(1,0) = src.dy(sx, sy);
00226                 double diff = dest(x, y) - src(sx, sy);
00227                 
00228                 m += outer(grad);
00229                 r -= diff*grad;
00230             }
00231         }
00232         
00233         linearSolve(m, r, s);
00234         
00235         matrix(0,2) -= s(0,0);
00236         matrix(1,2) -= s(1,0);
00237     }
00238 };

00334 template <class SrcIterator, class SrcAccessor, 
00335           class DestIterator, class DestAccessor, 
00336           int SPLINEORDER, class Functor>
00337 void 
00338 estimateAffineMotionImpl(SrcIterator sul, SrcIterator slr, SrcAccessor src,
00339                          DestIterator dul, DestIterator dlr, DestAccessor dest,
00340                          Matrix<double> & affineMatrix, 
00341                          AffineMotionEstimationOptions<SPLINEORDER> const & options,
00342                          Functor motionModel)
00343 {
00344     typedef typename NumericTraits<typename SrcAccessor::value_type>::RealPromote STmpType;
00345     typedef BasicImage<STmpType> STmpImage;
00346     typedef typename NumericTraits<typename DestAccessor::value_type>::RealPromote DTmpType;
00347     typedef BasicImage<DTmpType> DTmpImage;
00348     
00349     int toplevel = options.highest_level;
00350     ImagePyramid<STmpImage> srcPyramid(0, toplevel, sul, slr, src);
00351     ImagePyramid<DTmpImage> destPyramid(0, toplevel, dul, dlr, dest);
00352     
00353     if(options.use_laplacian_pyramid)
00354     {
00355         pyramidReduceBurtLaplacian(srcPyramid, 0, toplevel, options.burt_filter_strength);
00356         pyramidReduceBurtLaplacian(destPyramid, 0, toplevel, options.burt_filter_strength);
00357     }
00358     else
00359     {
00360         pyramidReduceBurtFilter(srcPyramid, 0, toplevel, options.burt_filter_strength);
00361         pyramidReduceBurtFilter(destPyramid, 0, toplevel, options.burt_filter_strength);
00362     }
00363         
00364     Matrix<double> currentMatrix(affineMatrix(2,2) == 0.0 
00365                                     ? identityMatrix<double>(3)
00366                                     : affineMatrix);
00367     currentMatrix(0,2) /= std::pow(2.0, toplevel);
00368     currentMatrix(1,2) /= std::pow(2.0, toplevel);
00369     
00370     for(int level = toplevel; level >= 0; --level)
00371     {
00372         SplineImageView<SPLINEORDER, STmpType> sp(srcImageRange(srcPyramid[level]));
00373         
00374         for(int iter = 0; iter < options.iterations_per_level; ++iter)
00375         {
00376             motionModel(sp, destPyramid[level], currentMatrix);
00377         }
00378         
00379         if(level > 0)
00380         {
00381             currentMatrix(0,2) *= 2.0;
00382             currentMatrix(1,2) *= 2.0;
00383         }
00384     }
00385     
00386     affineMatrix = currentMatrix;
00387 }
