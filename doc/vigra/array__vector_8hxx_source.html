00060 /** Provide STL conforming interface for C-arrays.
00061 
00062     This template implements much of the functionality of <a href="http://www.sgi.com/tech/stl/Vector.html">std::vector</a>
00063     on top of a C-array. <tt>ArrayVectorView</tt> does not manage the memory
00064     it refers to (i.e. it does not allocate or deallocate any memory).
00065     Thus, if the underlying memory changes, all dependent <tt>ArrayVectorView</tt>
00066     objects are invalidated. This is especially important when <tt>ArrayVectorView</tt>
00067     is used as a base class for <tt>ArrayVector</tt>, where several functions
00068     (e.g. resize(), insert()) can allocate new memory and thus invalidate the
00069     dependent views. The rules what operations invalidate view objects are the
00070     same as the rules concerning standard iterators.

00072     <b>\#include</b> <vigra/array_vector.hxx><br>

00396 template <class T>
00397 void
00398 ArrayVectorView <T>::copyImpl(const ArrayVectorView & rhs)
00399 {
00400     vigra_precondition (size() == rhs.size(),
00401         "ArrayVectorView::copy(): shape mismatch.");
00402     if(size() == 0)  // needed because MSVC debug assertions in std::copy() may fire  
00403         return;      // "invalid address: data_ == NULL" even when nothing is to be copied
00404     // use copy() or copy_backward() according to possible overlap of this and rhs
00405     if(data_ <= rhs.data())
00406     {
00407         std::copy(rhs.begin(), rhs.end(), begin());
00408     }
00409     else
00410     {
00411         std::copy_backward(rhs.begin(), rhs.end(), end());
00412     }
00413 }

00448 /** Replacement for <tt>std::vector</tt>.
00449 
00450     This template implements the same functionality as <tt>a href="http://www.sgi.com/tech/stl/Vector.html">std::vector</a></tt> (see there for detailed documentation).
00451     However, it gives two useful guarantees, that <tt>std::vector</tt> fails
00452     to provide:
00453 
00454     <ul>
00455     <li>The memory is always allocated as one contiguous piece.</li>
00456     <li>The iterator is always a <TT>T *</TT> </li>
00457     </ul>
00458 
00459     This means that memory managed by <tt>ArrayVector</tt> can be passed
00460     to algorithms that expect raw memory. This is especially important
00461     when legacy or C code has to be called, but it is also useful for certain
00462     optimizations.
00463 
00464     Moreover, <tt>ArrayVector</tt> is derived from <tt>ArrayVectorView</tt> so that one
00465     can create views of the array (in particular, subarrays). This implies another
00466     important difference to <tt>std::vector</tt>: the indexing operator
00467     (<tt>ArrayVector::operator[]</tt>) takes <tt>signed</tt> indices. In this way,
00468     an <tt>ArrayVectorView</tt> can be used with negative indices:
00469 
00470     \code
00471     ArrayVector<int> data(100);
00472     ArrayVectorView<int> view = data.subarray(50, 100);
00473 
00474     view[-50] = 1; // valid access
00475     \endcode
00476 
00477     Refer to the documentation of <tt>std::vector</tt> for a detailed
00478     description of <tt>ArrayVector</tt> functionality.
00479 
00480     <b>\#include</b> <vigra/array_vector.hxx><br>
00481     Namespace: vigra

00699 template <class T, class Alloc>
00700 typename ArrayVector<T, Alloc>::iterator
00701 ArrayVector<T, Alloc>::insert(iterator p, size_type n, value_type const & v)
00702 {
00703     difference_type pos = p - this->begin();
00704     size_type new_size = this->size() + n;
00705     if(new_size > capacity_)
00706     {
00707         size_type new_capacity = std::max(new_size, resizeFactor*capacity_);
00708         pointer new_data = reserve_raw(new_capacity);
00709         try
00710         {
00711             std::uninitialized_copy(this->begin(), p, new_data);
00712             std::uninitialized_fill(new_data + pos, new_data + pos + n, v);
00713             std::uninitialized_copy(p, this->end(), new_data + pos + n);
00714         }
00715         catch(...)
00716         {
00717             alloc_.deallocate(new_data, new_capacity);
00718             throw;
00719         }
