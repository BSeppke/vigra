00101 void
00102 getAxisPermutationImpl(ArrayVector<npy_intp> & permute,
00103                        python_ptr object, const char * name, 
00104                        AxisInfo::AxisType type, bool ignoreErrors)
00105 {
00106     python_ptr func(PyString_FromString(name), python_ptr::keep_count);
00107     python_ptr t(PyInt_FromLong((long)type), python_ptr::keep_count);
00108     python_ptr permutation(PyObject_CallMethodObjArgs(object, func.get(), t.get(), NULL), 
00109                            python_ptr::keep_count);
00110     if(!permutation && ignoreErrors)
00111     {
00112         PyErr_Clear();
00113         return;
00114     }
00115     pythonToCppException(permutation);
00116     
00117     if(!PySequence_Check(permutation))
00118     {
00119         if(ignoreErrors)
00120             return;
00121         std::string message = std::string(name) + "() did not return a sequence.";
00122         PyErr_SetString(PyExc_ValueError, message.c_str());
00123         pythonToCppException(false);
00124     }
00125         
00126     ArrayVector<npy_intp> res(PySequence_Length(permutation));
00127     for(int k=0; k<(int)res.size(); ++k)
00128     {
00129         python_ptr i(PySequence_GetItem(permutation, k), python_ptr::keep_count);
00130         if(!PyInt_Check(i))
00131         {
00132             if(ignoreErrors)
00133                 return;
00134             std::string message = std::string(name) + "() did not return a sequence of int.";
00135             PyErr_SetString(PyExc_ValueError, message.c_str());
00136             pythonToCppException(false);
00137         }
00138         res[k] = PyInt_AsLong(i);
00139     }
00140     res.swap(permute);
00141 }

00573     // transposeShape() means: only shape and resolution are transposed, not the axis keys

00579

00581                         ? 1

00595             newAxistags.setResolution(permute[k+tstart], axistags.resolution(permute[p[k]+tstart]));
00596         }
00597         shape = original_shape;
00598         axistags = newAxistags;
00599         
00600         return *this;
00601     }

00673                 original_shape.erase(original_shape.begin());
00674                 channelAxis = none;
00675             }
00676             break;
00677           case last:
00678             if(count > 0)
00679             {
00680                 shape[size()-1] = count;
00681             }
00682             else
00683             {
00684                 shape.pop_back();
00685                 original_shape.pop_back();
00686                 channelAxis = none;
00687             }
00688             break;
00689           case none:
00690             if(count > 0)
00691             {
00692                 shape.push_back(count);
00693                 original_shape.push_back(count);
00694                 channelAxis = last;
00695             }
00696             break;
00697         }
00698         return *this;
00699     }
00700     
00701     int channelCount() const
00702     {
00703         switch(channelAxis)
00704         {
00705           case first:
00706             return shape[0];
00707           case last:
00708             return shape[size()-1];
00709           default:
00710             return 1;
00711         }
00712     }
00713 };

00715 inline 
00716 void scaleAxisResolution(TaggedShape & tagged_shape)
00717 {
00718     if(tagged_shape.size() != tagged_shape.original_shape.size())
00719         return;
00720     
00721     int ntags = tagged_shape.axistags.size();
00722     
00723     ArrayVector<npy_intp> permute = tagged_shape.axistags.permutationToNormalOrder();
00724     
00725     int tstart = (tagged_shape.axistags.channelIndex(ntags) < ntags)
00726                     ? 1
00727                     : 0;
00728     int sstart = (tagged_shape.channelAxis == TaggedShape::first)
00729                     ? 1
00730                     : 0;
00731     int size = (int)tagged_shape.size() - sstart;
00732     
00733     for(int k=0; k<size; ++k)
00734     {
00735         int sk = k + sstart;
00736         if(tagged_shape.shape[sk] == tagged_shape.original_shape[sk])
00737             continue;
00738         double factor = (tagged_shape.original_shape[sk] - 1.0) / (tagged_shape.shape[sk] - 1.0);
00739         tagged_shape.axistags.scaleResolution(permute[k+tstart], factor);
00740     }
00741 }

00743 inline 
00744 void unifyTaggedShapeSize(TaggedShape & tagged_shape)
00745 {
00746     PyAxisTags axistags = tagged_shape.axistags;
00747     ArrayVector<npy_intp> & shape = tagged_shape.shape;
00748 
00749     int ndim = (int)shape.size();
00750     int ntags = axistags.size();
00751     
00752     long channelIndex = axistags.channelIndex();
00753 
00754     if(tagged_shape.channelAxis == TaggedShape::none)
00755     {
00756         // shape has no channel axis
00757         if(channelIndex == ntags)
00758         {
00759             // std::cerr << "branch (shape, axitags) 0 0\n";
00760             // axistags have no channel axis either => sizes should match
00761             vigra_precondition(ndim == ntags,
00762                  "constructArray(): size mismatch between shape and axistags.");
00763         }
00764         else
00765         {
00766             // std::cerr << "branch (shape, axitags) 0 1\n";
00767             if(ndim+1 == ntags)
00768             {
00769                 // std::cerr << "   drop channel axis\n";
00770                 // axistags have one additional element => drop the channel tag
00771                 // FIXME: would it be cleaner to make this an error ?
00772                 axistags.dropChannelAxis();
00773             }
00774             else
00775             {
00776                 vigra_precondition(ndim == ntags,
00777                      "constructArray(): size mismatch between shape and axistags.");
00778             }
00779         }
00780     }
00781     else
00782     {
00783         // shape has a channel axis
00784         if(channelIndex == ntags)
00785         {
00786             // std::cerr << "branch (shape, axitags) 1 0\n";
00787             // axistags have no channel axis => should be one element shorter
00788             vigra_precondition(ndim == ntags+1,
00789                  "constructArray(): size mismatch between shape and axistags.");
00790                  
00791             if(shape[0] == 1)
00792             {
00793                 // std::cerr << "   drop channel axis\n";
00794                 // we have a singleband image => drop the channel axis
00795                 shape.erase(shape.begin());
00796                 ndim -= 1;
00797             }
00798             else
00799             {
00800                 // std::cerr << "   insert channel axis\n";
00801                 // we have a multiband image => add a channel tag
00802                 axistags.insertChannelAxis();
00803             }
00804         }
00805         else
00806         {
00807             // std::cerr << "branch (shape, axitags) 1 1\n";
00808             // axistags have channel axis => sizes should match
00809             vigra_precondition(ndim == ntags,
00810                  "constructArray(): size mismatch between shape and axistags.");
00811         }
00812     }
00813 }

00815 inline
00816 ArrayVector<npy_intp> finalizeTaggedShape(TaggedShape & tagged_shape)
00817 {
00818     if(tagged_shape.axistags)
00819     {
00820         tagged_shape.rotateToNormalOrder();
00821     
00822         // we assume here that the axistag object belongs to the array to be created
00823         // so that we can freely edit it
00824         scaleAxisResolution(tagged_shape);
00825             
00826         // this must be after scaleAxisResolution(), because the latter requires 
00827         // shape and original_shape to be still in sync
00828         unifyTaggedShapeSize(tagged_shape);
00829                 
00830         if(tagged_shape.channelDescription != "")
00831             tagged_shape.axistags.setChannelDescription(tagged_shape.channelDescription);
00832     }
00833     return tagged_shape.shape;
00834 }
