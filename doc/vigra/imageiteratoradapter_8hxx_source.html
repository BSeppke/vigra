00506         /** Construct from an the image iterator to be adapted.
00507         */
00508     LineIterator(IMAGE_ITERATOR  const & start,
00509                  IMAGE_ITERATOR  const & end)
00510     : IMAGE_ITERATOR(start), x_(0.0), y_(0.0)
00511     {
00512         int dx = end.x - start.x;
00513         int dy = end.y - start.y;
00514         int adx = (dx < 0) ? -dx : dx;
00515         int ady = (dy < 0) ? -dy : dy;
00516         int dd = (adx > ady) ? adx : ady;
00517         if(dd == 0) dd = 1;
00518 
00519         dx_ = (double)dx / dd;
00520         dy_ = (double)dy / dd;
00521         if(adx > ady) y_ += dy_ / 2.0;
00522         else          x_ += dx_ / 2.0;
00523     }
00524 
00525     /** @name Navigation */
00526     //@{
00527         ///
00528     LineIterator &  operator++()
00529     {
00530         x_ += dx_;

00532             x_ -= 1.0;
00533             ++(this->x);
00534         }
00535         else if(x_ <= -1.0) {
00536             x_ += 1.0;
00537             --(this->x);
00538         }
00539         y_ += dy_;
00540         if(y_ >= 1.0) {
00541             y_ -= 1.0;
00542             ++(this->y);
00543         }
00544         else if(y_ <= -1.0) {

00546             --(this->y);
00547         }
00548         return *this;
