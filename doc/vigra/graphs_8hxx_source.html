00094 // tie() support for std::pair, similar to Boost's one:
00095 // (necessary because std::pair doesn't define a suitable assignment operator)
00096 template<class T1, class T2>
00097 class tie_adapter 
00098 {
00099   public:
00100     tie_adapter(T1 &x, T2 &y) 
00101         : x_(x), y_(y)
00102     {}
00103 
00104     template<class X, class Y>
00105     tie_adapter & operator=(const std::pair<X, Y> &pair)
00106     {
00107         x_ = pair.first;
00108         y_ = pair.second;
00109         return *this;
00110     }
00111     
00112   protected:
00113     T1 &x_;
00114     T2 &y_;
00115 };
00116 
00117 template<class T1, class T2>
00118 inline
00119 tie_adapter<T1, T2>
00120 tie(T1& t1, T2& t2) 
00121 {
00122     return tie_adapter<T1, T2>(t1, t2);
