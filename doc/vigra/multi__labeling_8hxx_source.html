00051 template <class Graph, class T1Map, class T2Map, class Equal>
00052 typename T2Map::value_type
00053 labelGraph(Graph const & g, 
00054            T1Map const & data,
00055            T2Map & labels,
00056            Equal const & equal)
00057 {
00058     typedef typename Graph::NodeIt        graph_scanner;
00059     typedef typename Graph::OutBackArcIt  neighbor_iterator;
00060     typedef typename T2Map::value_type    LabelType;
00061 
00062     vigra::detail::UnionFindArray<LabelType>  regions;
00063 
00064     // pass 1: find connected components
00065     for (graph_scanner node(g); node != INVALID; ++node) 
00066     {
00067         typename T1Map::value_type center = data[*node];
00068         
00069         // define tentative label for current node
00070         LabelType currentLabel = regions.nextFreeLabel();
00071         
00072         for (neighbor_iterator arc(g, node); arc != INVALID; ++arc)
00073         {
00074             // merge regions if colors are equal
00075             if(equal(center, data[g.target(*arc)]))
00076             {
00077                 LabelType neighborLabel = regions[labels[g.target(*arc)]];
00078                 currentLabel = regions.makeUnion(neighborLabel, currentLabel);
00079             }
00080         }
00081         // set label of current node
00082         labels[*node] = regions.finalizeLabel(currentLabel);
00083     }
00084     
00085     LabelType count = regions.makeContiguous();
00086 
00087     // pass 2: make component labels contiguous
00088     for (graph_scanner node(g); node != INVALID; ++node) 
00089     {
00090         labels[*node] = regions[labels[*node]];
00091     }
00092     return count;
00093 }

00155 /** \brief Find the connected components of a MultiArray with arbitrary many dimensions.
00156 
00157     <b> Declaration:</b>
00158 
00159     \code
00160     namespace vigra {
00161 
00162         template <unsigned int N, class T, class S1,
00163                                   class Label, class S2,
00164                   class EqualityFunctor = std::equal_to<T> >
00165         Label 
00166         labelMultiArray(MultiArrayView<N, T, S1> const & data,
00167                         MultiArrayView<N, Label, S2> labels,
00168                         NeighborhoodType neighborhood = DirectNeighborhood,
00169                         EqualityFunctor equal = std::equal_to<T>())
00170 
00171     }
00172     \endcode
00173 
00174     Connected components are defined as regions with uniform values. 
00175     Thus, the value type <tt>T</tt> of the input array \a data either 
00176     must be equality comparable, or an EqualityFunctor must be
00177     provided that realizes the desired predicate. The destination 
00178     array's value type <tt>Label</tt> should be large enough to hold 
00179     the labels without overflow. Region numbers will form a consecutive 
00180     sequence starting at <b>one</b> and ending with the region number 
00181     returned by the function (inclusive).
00182     
00183     Argument \a neighborhood specifies the type of connectivity used. It can
00184     take the values <tt>DirectNeighborhood</tt> (which corresponds to
00185     4-neighborhood in 2D and 6-neighborhood in 3D, default) or
00186     <tt>IndirectNeighborhood</tt> (which corresponds to
00187     8-neighborhood in 2D and 26-neighborhood in 3D).
00188 
00189     Return:  the number of regions found (= highest region label, because labeling starts at 1)
00190 
00191     <b> Usage:</b>
00192 
00193     <b>\#include</b> <vigra/multi_labeling.hxx><br>
00194     Namespace: vigra
00195 
00196     \code
00197     MultiArray<3,int> src(Shape3(w,h,d));
00198     MultiArray<3,int> dest(Shape3(w,h,d));
00199     
00200     // find 6-connected regions
00201     int max_region_label = labelMultiArray(src, dest);
00202 
00203     // find 26-connected regions
00204     max_region_label = labelMultiArray(src, dest, IndirectNeighborhood);
00205     \endcode
00206 
00207     <b> Required Interface:</b>
00208 
00209     \code
00210     T t;
00211     t == t                      // T is equality comparable
00212 
00213     EqualityFunctor equal;      // or a suitable functor is supplied
00214     equal(t, t)
00215     \endcode
00216 */

00266                                       T backgroundValue = T(),
00267                                       Equal equal = std::equal<T>());
00268 
00269     }
00270     \endcode
00271 
00272     This function is the same as \ref labelMultiArray(), except for 
00273     the additional parameter \a backgroundValue. Points in the input
00274     array \a data with this value (which default to zero) are ignored 
00275     during labeling, and  their output label is automatically set to 
00276     zero. Region numbers will be a consecutive sequence starting at 
00277     zero (when background was present) or at one (when no background 
00278     was present) and ending with the region number returned by the

00282     because background has label 0)

00285 
00286     <b>\#include</b> <vigra/multi_labeling.hxx><br>
00287     Namespace: vigra
00288

00290     MultiArray<3, int> src(Shape3(w,h,d));
00291     MultiArray<3, int> dest(Shape3(w,h,d));
00292     
00293     // find 6-connected regions, ignoring background value zero (the default)
00294     int max_region_label = labelMultiArrayWithBackground(src, dest);
00295 
00296     // find 26-connected regions, using -1 as the background value
00297     max_region_label = labelMultiArrayWithBackground(src, dest, IndirectNeighborhood, -1);
00298     \endcode
00299 
00300     <b> Required Interface:</b>

00302     \code
